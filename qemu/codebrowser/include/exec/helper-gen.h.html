9#define DEF_HELPER_FLAGS_0(name, flags, ret) \
10static inline void glue(gen_helper_, name)(dh_retvar_decl0(ret)) \
11{ \
12 tcg_gen_callN(&tcg_ctx, HELPER(name), dh_retvar(ret), 0, NULL); \
13}
14
15#define DEF_HELPER_FLAGS_1(name, flags, ret, t1) \
16static inline void glue(gen_helper_, name)(dh_retvar_decl(ret) \
17 dh_arg_decl(t1, 1)) \
18{ \
19 TCGArg args[1] = { dh_arg(t1, 1) }; \
20 tcg_gen_callN(&tcg_ctx, HELPER(name), dh_retvar(ret), 1, args); \
21}
