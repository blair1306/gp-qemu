20#ifndef QEMU_TB_CONTEXT_H
21#define QEMU_TB_CONTEXT_H
22
23#include "qemu/thread.h"
24#include "qemu/qht.h"
25
26#define CODE_GEN_HTABLE_BITS 15
27#define CODE_GEN_HTABLE_SIZE (1 << CODE_GEN_HTABLE_BITS)
28
29typedef struct TranslationBlock TranslationBlock;
30typedef struct TBContext TBContext;
31
32struct TBContext {
33
34 TranslationBlock **tbs;
35 struct qht htable;
36 size_t tbs_size;
37 int nb_tbs;
38 /* any access to the tbs or the page table must use this lock */
39 QemuMutex tb_lock;
40
41 /* statistics */
42 unsigned tb_flush_count;
43 int tb_phys_invalidate_count;
44};
45
46#endif
