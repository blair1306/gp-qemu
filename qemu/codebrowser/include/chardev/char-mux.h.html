24#ifndef CHAR_MUX_H
25#define CHAR_MUX_H
26
27#include "chardev/char.h"
28#include "chardev/char-fe.h"
29
30extern bool muxes_realized;
31
32#define MAX_MUX 4
33#define MUX_BUFFER_SIZE 32 /* Must be a power of 2. */
34#define MUX_BUFFER_MASK (MUX_BUFFER_SIZE - 1)
35typedef struct MuxChardev {
36 Chardev parent;
37 CharBackend *backends[MAX_MUX];
38 CharBackend chr;
39 int focus;
40 int mux_cnt;
41 int term_got_escape;
42 int max_size;
43 /* Intermediate input buffer catches escape sequences even if the
44 currently active device is not accepting any input - but only until it
45 is full as well. */
46 unsigned char buffer[MAX_MUX][MUX_BUFFER_SIZE];
47 int prod[MAX_MUX];
48 int cons[MAX_MUX];
49 int timestamps;
50
51 /* Protected by the Chardev chr_write_lock. */
52 int linestart;
53 int64_t timestamps_start;
54} MuxChardev;
55
56#define MUX_CHARDEV(obj) OBJECT_CHECK(MuxChardev, (obj), TYPE_CHARDEV_MUX)
57#define CHARDEV_IS_MUX(chr) \
58 object_dynamic_cast(OBJECT(chr), TYPE_CHARDEV_MUX)
59
60void mux_chr_set_handlers(Chardev *chr, GMainContext *context);
61void mux_set_focus(Chardev *chr, int focus);
62void mux_chr_send_all_event(Chardev *chr, int event);
63
64#endif /* CHAR_MUX_H */
