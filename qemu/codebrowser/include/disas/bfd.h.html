1/* Interface between the opcode library and its callers.
2 Written by Cygnus Support, 1993.
3
4 The opcode library (libopcodes.a) provides instruction decoders for
5 a large variety of instruction sets, callable with an identical
6 interface, for making instruction-processing programs more independent
7 of the instruction set being processed. */

257/* This struct is passed into the instruction decoding routine,
258 and is passed back out into each callback. The various fields are used
259 for conveying information from your main routine into your callbacks,
260 for passing information into the instruction decoders (such as the
261 addresses of the callback functions), or for passing information
262 back from the instruction decoders to their callers.
263
264 It must be initialized before it is first passed; this can be done
265 by hand, or using one of the initialization macros below. */
266
267typedef struct disassemble_info {
268 fprintf_function fprintf_func;
269 FILE *stream;
270 PTR application_data;
271
272 /* Target description. We could replace this with a pointer to the bfd,
273 but that would require one. There currently isn't any such requirement
274 so to avoid introducing one we record these explicitly. */
275 /* The bfd_flavour. This can be bfd_target_unknown_flavour. */
276 enum bfd_flavour flavour;
277 /* The bfd_arch value. */
278 enum bfd_architecture arch;
279 /* The bfd_mach value. */
280 unsigned long mach;
281 /* Endianness (for bi-endian cpus). Mono-endian cpus can ignore this. */
282 enum bfd_endian endian;
283
284 /* An array of pointers to symbols either at the location being disassembled
285 or at the start of the function being disassembled. The array is sorted
286 so that the first symbol is intended to be the one used. The others are
287 present for any misc. purposes. This is not set reliably, but if it is
288 not NULL, it is correct. */
289 asymbol **symbols;
290 /* Number of symbols in array. */
291 int num_symbols;
292
293 /* For use by the disassembler.
294 The top 16 bits are reserved for public use (and are documented here).
295 The bottom 16 bits are for the internal use of the disassembler. */
296 unsigned long flags;
297#define INSN_HAS_RELOC 0x80000000
298#define INSN_ARM_BE32 0x00010000
299 PTR private_data;
300
301 /* Function used to get bytes to disassemble. MEMADDR is the
302 address of the stuff to be disassembled, MYADDR is the address to
303 put the bytes in, and LENGTH is the number of bytes to read.
304 INFO is a pointer to this struct.
305 Returns an errno value or 0 for success. */
306 int (*read_memory_func)
307 (bfd_vma memaddr, bfd_byte *myaddr, int length,
308 struct disassemble_info *info);
309
310 /* A place to stash the real read_memory_func if read_memory_func wants to
311 do some funky address arithmetic or similar (e.g. for ARM BE32 mode). */
312 int (*read_memory_inner_func)
313 (bfd_vma memaddr, bfd_byte *myaddr, int length,
314 struct disassemble_info *info);
315
316 /* Function which should be called if we get an error that we can't
317 recover from. STATUS is the errno value from read_memory_func and
318 MEMADDR is the address that we were trying to read. INFO is a
319 pointer to this struct. */
320 void (*memory_error_func)
321 (int status, bfd_vma memaddr, struct disassemble_info *info);
322
323 /* Function called to print ADDR. */
324 void (*print_address_func)
325 (bfd_vma addr, struct disassemble_info *info);
326
327 /* Function called to print an instruction. The function is architecture
328 * specific.
329 */
330 int (*print_insn)(bfd_vma addr, struct disassemble_info *info);
331
332 /* Function called to determine if there is a symbol at the given ADDR.
333 If there is, the function returns 1, otherwise it returns 0.
334 This is used by ports which support an overlay manager where
335 the overlay number is held in the top part of an address. In
336 some circumstances we want to include the overlay number in the
337 address, (normally because there is a symbol associated with
338 that address), but sometimes we want to mask out the overlay bits. */
339 int (* symbol_at_address_func)
340 (bfd_vma addr, struct disassemble_info * info);
341
342 /* These are for buffer_read_memory. */
343 bfd_byte *buffer;
344 bfd_vma buffer_vma;
345 int buffer_length;
346
347 /* This variable may be set by the instruction decoder. It suggests
348 the number of bytes objdump should display on a single line. If
349 the instruction decoder sets this, it should always set it to
350 the same value in order to get reasonable looking output. */
351 int bytes_per_line;
352
353 /* the next two variables control the way objdump displays the raw data */
354 /* For example, if bytes_per_line is 8 and bytes_per_chunk is 4, the */
355 /* output will look like this:
356 00: 00000000 00000000
357 with the chunks displayed according to "display_endian". */
358 int bytes_per_chunk;
359 enum bfd_endian display_endian;
360
361 /* Results from instruction decoders. Not all decoders yet support
362 this information. This info is set each time an instruction is
363 decoded, and is only valid for the last such instruction.
364
365 To determine whether this decoder supports this information, set
366 insn_info_valid to 0, decode an instruction, then check it. */
367
368 char insn_info_valid; /* Branch info has been set. */
369 char branch_delay_insns; /* How many sequential insn's will run before
370 a branch takes effect. (0 = normal) */
371 char data_size; /* Size of data reference in insn, in bytes */
372 enum dis_insn_type insn_type; /* Type of instruction */
373 bfd_vma target; /* Target address of branch or dref, if known;
374 zero if unknown. */
375 bfd_vma target2; /* Second target address for dref2 */
376
377 /* Command line options specific to the target disassembler. */
378 char * disassembler_options;
379
380} disassemble_info;

458/* Macro to initialize a disassemble_info struct. This should be called
459 by all applications creating such a struct. */
460#define INIT_DISASSEMBLE_INFO(INFO, STREAM, FPRINTF_FUNC) \
461 (INFO).flavour = bfd_target_unknown_flavour, \
462 (INFO).arch = bfd_arch_unknown, \
463 (INFO).mach = 0, \
464 (INFO).endian = BFD_ENDIAN_UNKNOWN, \
465 INIT_DISASSEMBLE_INFO_NO_ARCH(INFO, STREAM, FPRINTF_FUNC)
466
467/* Call this macro to initialize only the internal variables for the
468 disassembler. Architecture dependent things such as byte order, or machine
469 variant are not touched by this macro. This makes things much easier for
470 GDB which must initialize these things separately. */
471
472#define INIT_DISASSEMBLE_INFO_NO_ARCH(INFO, STREAM, FPRINTF_FUNC) \
473 (INFO).fprintf_func = (FPRINTF_FUNC), \
474 (INFO).stream = (STREAM), \
475 (INFO).symbols = NULL, \
476 (INFO).num_symbols = 0, \
477 (INFO).private_data = NULL, \
478 (INFO).buffer = NULL, \
479 (INFO).buffer_vma = 0, \
480 (INFO).buffer_length = 0, \
481 (INFO).read_memory_func = buffer_read_memory, \
482 (INFO).memory_error_func = perror_memory, \
483 (INFO).print_address_func = generic_print_address, \
484 (INFO).print_insn = NULL, \
485 (INFO).symbol_at_address_func = generic_symbol_at_address, \
486 (INFO).flags = 0, \
487 (INFO).bytes_per_line = 0, \
488 (INFO).bytes_per_chunk = 0, \
