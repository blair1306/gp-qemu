99static inline bool qemu_using_spice(Error **errp)
100{
101 if (!using_spice) {
102 error_set(errp, ERROR_CLASS_DEVICE_NOT_ACTIVE,
103 "SPICE is not in use");
104 return false;
105 }
106 return true;
107}
