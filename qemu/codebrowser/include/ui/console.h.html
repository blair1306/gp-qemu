345extern chtype vga_to_curses[];
346#else
347typedef unsigned long console_ch_t;
348#endif
349static inline void console_write_ch(console_ch_t *dest, uint32_t ch)
350{
351 uint8_t c = ch;
352#ifdef CONFIG_CURSES
353 if (vga_to_curses[c]) {
354 ch &= ~(console_ch_t)0xff;
355 ch |= vga_to_curses[c];
356 }
357#else
358 if (c == '\0') {
359 ch |= ' ';
360 }
361#endif
362 *dest = ch;
363}
