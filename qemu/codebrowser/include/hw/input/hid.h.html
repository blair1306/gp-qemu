11typedef struct HIDPointerEvent {
12 int32_t xdx, ydy; /* relative iff it's a mouse, otherwise absolute */
13 int32_t dz, buttons_state;
14} HIDPointerEvent;

16#define QUEUE_LENGTH 16 /* should be enough for a triple-click */
17#define QUEUE_MASK (QUEUE_LENGTH-1u)
18#define QUEUE_INCR(v) ((v)++, (v) &= QUEUE_MASK)
19
20typedef struct HIDState HIDState;
21typedef void (*HIDEventFunc)(HIDState *s);
22
23typedef struct HIDMouseState {
24 HIDPointerEvent queue[QUEUE_LENGTH];
25 int mouse_grabbed;
26} HIDMouseState;
27
28typedef struct HIDKeyboardState {
29 uint32_t keycodes[QUEUE_LENGTH];
30 uint16_t modifiers;
31 uint8_t leds;
32 uint8_t key[16];
33 int32_t keys;
34} HIDKeyboardState;
35
36struct HIDState {
37 union {
38 HIDMouseState ptr;
39 HIDKeyboardState kbd;
40 };
41 uint32_t head; /* index into circular queue */
42 uint32_t n;
43 int kind;
44 int32_t protocol;
45 uint8_t idle;
46 bool idle_pending;
47 QEMUTimer *idle_timer;
48 HIDEventFunc event;
49 QemuInputHandlerState *s;
50};
