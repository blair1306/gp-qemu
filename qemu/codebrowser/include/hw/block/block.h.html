16/* Configuration */
17
18typedef struct BlockConf {
19 BlockBackend *blk;
20 uint16_t physical_block_size;
21 uint16_t logical_block_size;
22 uint16_t min_io_size;
23 uint32_t opt_io_size;
24 int32_t bootindex;
25 uint32_t discard_granularity;
26 /* geometry, not all devices use this */
27 uint32_t cyls, heads, secs;
28 OnOffAuto wce;
29 bool share_rw;
30 BlockdevOnError rerror;
31 BlockdevOnError werror;
32} BlockConf;
33
34static inline unsigned int get_physical_block_exp(BlockConf *conf)
35{
36 unsigned int exp = 0, size;
37
38 for (size = conf->physical_block_size;
39 size > conf->logical_block_size;
40 size >>= 1) {
41 exp++;
42 }
43
44 return exp;
45}
