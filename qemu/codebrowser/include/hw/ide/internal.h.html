535static inline uint8_t ide_dma_cmd_to_retry(uint8_t dma_cmd)
536{
537 switch (dma_cmd) {
538 case IDE_DMA_READ:
539 return IDE_RETRY_DMA | IDE_RETRY_READ;
540 case IDE_DMA_WRITE:
541 return IDE_RETRY_DMA;
542 case IDE_DMA_TRIM:
543 return IDE_RETRY_DMA | IDE_RETRY_TRIM;
544 case IDE_DMA_ATAPI:
545 return IDE_RETRY_ATAPI;
546 default:
547 break;
548 }
549 return 0;
550}
