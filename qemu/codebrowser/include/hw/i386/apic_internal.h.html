214static inline void apic_set_bit(uint32_t *tab, int index)
215{
216 int i, mask;
217 i = index >> 5;
218 mask = 1 << (index & 0x1f);
219 tab[i] |= mask;
220}
221
222static inline int apic_get_bit(uint32_t *tab, int index)
223{
224 int i, mask;
225 i = index >> 5;
226 mask = 1 << (index & 0x1f);
227 return !!(tab[i] & mask);
228}
