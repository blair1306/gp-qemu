25#ifndef HW_PCSPK_H
26#define HW_PCSPK_H
27
28#include "hw/hw.h"
29#include "hw/isa/isa.h"
30
31#define TYPE_PC_SPEAKER "isa-pcspk"
32
33static inline ISADevice *pcspk_init(ISABus *bus, ISADevice *pit)
34{
35 DeviceState *dev;
36 ISADevice *isadev;
37
38 isadev = isa_create(bus, TYPE_PC_SPEAKER);
39 dev = DEVICE(isadev);
40 qdev_prop_set_uint32(dev, "iobase", 0x61);
41 object_property_set_link(OBJECT(dev), OBJECT(pit), "pit", NULL);
42 qdev_init_nofail(dev);
43
44 return isadev;
45}
46
47#endif /* HW_PCSPK_H */
