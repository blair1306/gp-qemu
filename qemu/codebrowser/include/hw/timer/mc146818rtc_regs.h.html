68static inline uint32_t periodic_period_to_clock(int period_code)
69{
70 if (!period_code) {
71 return 0;
72 }
73
74 if (period_code <= 2) {
75 period_code += 7;
76 }
77 /* period in 32 Khz cycles */
78 return 1 << (period_code - 1);
79}
80
81#define RTC_CLOCK_RATE 32768
82
83static inline int64_t periodic_clock_to_ns(int64_t clocks)
84{
85 return muldiv64(clocks, NANOSECONDS_PER_SECOND, RTC_CLOCK_RATE);
86}
