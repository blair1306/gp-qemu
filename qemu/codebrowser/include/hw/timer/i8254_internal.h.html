25#ifndef QEMU_I8254_INTERNAL_H
26#define QEMU_I8254_INTERNAL_H
27
28#include "hw/hw.h"
29#include "hw/i386/pc.h"
30#include "hw/isa/isa.h"
31
32typedef struct PITChannelState {
33 int count; /* can be 65536 */
34 uint16_t latched_count;
35 uint8_t count_latched;
36 uint8_t status_latched;
37 uint8_t status;
38 uint8_t read_state;
39 uint8_t write_state;
40 uint8_t write_latch;
41 uint8_t rw_mode;
42 uint8_t mode;
43 uint8_t bcd; /* not supported */
44 uint8_t gate; /* timer start */
45 int64_t count_load_time;
46 /* irq handling */
47 int64_t next_transition_time;
48 QEMUTimer *irq_timer;
49 qemu_irq irq;
50 uint32_t irq_disabled;
51} PITChannelState;
52
53typedef struct PITCommonState {
54 ISADevice dev;
55 MemoryRegion ioports;
56 uint32_t iobase;
57 PITChannelState channels[3];
58} PITCommonState;
59
60typedef struct PITCommonClass {
61 ISADeviceClass parent_class;
62
63 void (*set_channel_gate)(PITCommonState *s, PITChannelState *sc, int val);
64 void (*get_channel_info)(PITCommonState *s, PITChannelState *sc,
65 PITChannelInfo *info);
66 void (*pre_save)(PITCommonState *s);
67 void (*post_load)(PITCommonState *s);
68} PITCommonClass;
69
70int pit_get_out(PITChannelState *s, int64_t current_time);
71int64_t pit_get_next_transition_time(PITChannelState *s, int64_t current_time);
72void pit_get_channel_info_common(PITCommonState *s, PITChannelState *sc,
73 PITChannelInfo *info);
74void pit_reset_common(PITCommonState *s);
75
76#endif /* QEMU_I8254_INTERNAL_H */
