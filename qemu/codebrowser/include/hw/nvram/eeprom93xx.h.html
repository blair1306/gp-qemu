20#ifndef EEPROM93XX_H
21#define EEPROM93XX_H
22
23typedef struct _eeprom_t eeprom_t;
24
25/* Create a new EEPROM with (nwords * 2) bytes. */
26eeprom_t *eeprom93xx_new(DeviceState *dev, uint16_t nwords);
27
28/* Destroy an existing EEPROM. */
29void eeprom93xx_free(DeviceState *dev, eeprom_t *eeprom);
30
31/* Read from the EEPROM. */
32uint16_t eeprom93xx_read(eeprom_t *eeprom);
33
34/* Write to the EEPROM. */
35void eeprom93xx_write(eeprom_t *eeprom, int eecs, int eesk, int eedi);
36
37/* Get EEPROM data array. */
38uint16_t *eeprom93xx_data(eeprom_t *eeprom);
39
40#endif /* EEPROM93XX_H */
