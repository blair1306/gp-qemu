25#ifndef SDHCI_H
26#define SDHCI_H
27
28#include "qemu-common.h"
29#include "hw/block/block.h"
30#include "hw/pci/pci.h"
31#include "hw/sysbus.h"
32#include "hw/sd/sd.h"
33
34/* SD/MMC host controller state */
35typedef struct SDHCIState {
36 union {
37 PCIDevice pcidev;
38 SysBusDevice busdev;
39 };
40 SDBus sdbus;
41 MemoryRegion iomem;
42
43 QEMUTimer *insert_timer; /* timer for 'changing' sd card. */
44 QEMUTimer *transfer_timer;
45 qemu_irq eject_cb;
46 qemu_irq ro_cb;
47 qemu_irq irq;
48
49 uint32_t sdmasysad; /* SDMA System Address register */
50 uint16_t blksize; /* Host DMA Buff Boundary and Transfer BlkSize Reg */
51 uint16_t blkcnt; /* Blocks count for current transfer */
52 uint32_t argument; /* Command Argument Register */
53 uint16_t trnmod; /* Transfer Mode Setting Register */
54 uint16_t cmdreg; /* Command Register */
55 uint32_t rspreg[4]; /* Response Registers 0-3 */
56 uint32_t prnsts; /* Present State Register */
57 uint8_t hostctl; /* Host Control Register */
58 uint8_t pwrcon; /* Power control Register */
59 uint8_t blkgap; /* Block Gap Control Register */
60 uint8_t wakcon; /* WakeUp Control Register */
61 uint16_t clkcon; /* Clock control Register */
62 uint8_t timeoutcon; /* Timeout Control Register */
63 uint8_t admaerr; /* ADMA Error Status Register */
64 uint16_t norintsts; /* Normal Interrupt Status Register */
65 uint16_t errintsts; /* Error Interrupt Status Register */
66 uint16_t norintstsen; /* Normal Interrupt Status Enable Register */
67 uint16_t errintstsen; /* Error Interrupt Status Enable Register */
68 uint16_t norintsigen; /* Normal Interrupt Signal Enable Register */
69 uint16_t errintsigen; /* Error Interrupt Signal Enable Register */
70 uint16_t acmd12errsts; /* Auto CMD12 error status register */
71 uint64_t admasysaddr; /* ADMA System Address Register */
72
73 uint32_t capareg; /* Capabilities Register */
74 uint32_t maxcurr; /* Maximum Current Capabilities Register */
75 uint8_t *fifo_buffer; /* SD host i/o FIFO buffer */
76 uint32_t buf_maxsz;
77 uint16_t data_count; /* current element in FIFO buffer */
78 uint8_t stopped_state;/* Current SDHC state */
79 bool pending_insert_quirk;/* Quirk for Raspberry Pi card insert int */
80 bool pending_insert_state;
81 /* Buffer Data Port Register - virtual access point to R and W buffers */
82 /* Software Reset Register - always reads as 0 */
83 /* Force Event Auto CMD12 Error Interrupt Reg - write only */
84 /* Force Event Error Interrupt Register- write only */
85 /* RO Host Controller Version Register always reads as 0x2401 */
86} SDHCIState;
87
88#define TYPE_PCI_SDHCI "sdhci-pci"
89#define PCI_SDHCI(obj) OBJECT_CHECK(SDHCIState, (obj), TYPE_PCI_SDHCI)
90
91#define TYPE_SYSBUS_SDHCI "generic-sdhci"
92#define SYSBUS_SDHCI(obj) \
93 OBJECT_CHECK(SDHCIState, (obj), TYPE_SYSBUS_SDHCI)
94
95#endif /* SDHCI_H */
