31#define SELF_ANNOUNCE_ROUNDS 5
32
33static inline
34int64_t self_announce_delay(int round)
35{
36 assert(round < SELF_ANNOUNCE_ROUNDS && round > 0);
37 /* delay 50ms, 150ms, 250ms, ... */
38 return 50 + (SELF_ANNOUNCE_ROUNDS - round - 1) * 100;
39}
