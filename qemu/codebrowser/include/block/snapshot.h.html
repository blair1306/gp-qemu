38typedef struct QEMUSnapshotInfo {
39 char id_str[128]; /* unique snapshot id */
40 /* the following fields are informative. They are not needed for
41 the consistency of the snapshot */
42 char name[256]; /* user chosen name */
43 uint64_t vm_state_size; /* VM state info size */
44 uint32_t date_sec; /* UTC date of the snapshot */
45 uint32_t date_nsec;
46 uint64_t vm_clock_nsec; /* VM clock relative to boot */
47} QEMUSnapshotInfo;
