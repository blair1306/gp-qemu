25#ifndef BLOCK_ACCOUNTING_H
26#define BLOCK_ACCOUNTING_H
27
28#include "qemu/timed-average.h"
29#include "qemu/thread.h"
30
31typedef struct BlockAcctTimedStats BlockAcctTimedStats;
32typedef struct BlockAcctStats BlockAcctStats;
33
34enum BlockAcctType {
35 BLOCK_ACCT_READ,
36 BLOCK_ACCT_WRITE,
37 BLOCK_ACCT_FLUSH,
38 BLOCK_MAX_IOTYPE,
39};
40
41struct BlockAcctTimedStats {
42 BlockAcctStats *stats;
43 TimedAverage latency[BLOCK_MAX_IOTYPE];
44 unsigned interval_length; /* in seconds */
45 QSLIST_ENTRY(BlockAcctTimedStats) entries;
46};
47
48struct BlockAcctStats {
49 QemuMutex lock;
50 uint64_t nr_bytes[BLOCK_MAX_IOTYPE];
51 uint64_t nr_ops[BLOCK_MAX_IOTYPE];
52 uint64_t invalid_ops[BLOCK_MAX_IOTYPE];
53 uint64_t failed_ops[BLOCK_MAX_IOTYPE];
54 uint64_t total_time_ns[BLOCK_MAX_IOTYPE];
55 uint64_t merged[BLOCK_MAX_IOTYPE];
56 int64_t last_access_time_ns;
57 QSLIST_HEAD(, BlockAcctTimedStats) intervals;
58 bool account_invalid;
59 bool account_failed;
60};
61
62typedef struct BlockAcctCookie {
63 int64_t bytes;
64 int64_t start_time_ns;
65 enum BlockAcctType type;
66} BlockAcctCookie;
67
68void block_acct_init(BlockAcctStats *stats);
69void block_acct_setup(BlockAcctStats *stats, bool account_invalid,
70 bool account_failed);
71void block_acct_cleanup(BlockAcctStats *stats);
72void block_acct_add_interval(BlockAcctStats *stats, unsigned interval_length);
73BlockAcctTimedStats *block_acct_interval_next(BlockAcctStats *stats,
74 BlockAcctTimedStats *s);
75void block_acct_start(BlockAcctStats *stats, BlockAcctCookie *cookie,
76 int64_t bytes, enum BlockAcctType type);
77void block_acct_done(BlockAcctStats *stats, BlockAcctCookie *cookie);
78void block_acct_failed(BlockAcctStats *stats, BlockAcctCookie *cookie);
79void block_acct_invalid(BlockAcctStats *stats, enum BlockAcctType type);
80void block_acct_merge_done(BlockAcctStats *stats, enum BlockAcctType type,
81 int num_requests);
82int64_t block_acct_idle_time_ns(BlockAcctStats *stats);
83double block_acct_queue_depth(BlockAcctTimedStats *stats,
84 enum BlockAcctType type);
85
86#endif
