264/*
265 * Based on values from <linux/cdrom.h> but extending CD_MINS
266 * to the maximum common size allowed by the Orange's Book ATIP
267 *
268 * 90 and 99 min CDs are also available but using them as the
269 * upper limit reduces the effectiveness of the heuristic to
270 * detect DVDs burned to less than 25% of their maximum capacity
271 */
272
273/* Some generally useful CD-ROM information */
274#define CD_MINS 80 /* max. minutes per CD */
275#define CD_SECS 60 /* seconds per minute */
276#define CD_FRAMES 75 /* frames per second */
277#define CD_FRAMESIZE 2048 /* bytes per frame, "cooked" mode */
278#define CD_MAX_BYTES (CD_MINS * CD_SECS * CD_FRAMES * CD_FRAMESIZE)
279#define CD_MAX_SECTORS (CD_MAX_BYTES / 512)
