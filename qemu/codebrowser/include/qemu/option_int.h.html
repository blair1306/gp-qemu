29#include "qemu/option.h"

32struct QemuOpt {
33 char *name;
34 char *str;
35
36 const QemuOptDesc *desc;
37 union {
38 bool boolean;
39 uint64_t uint;
40 } value;
41
42 QemuOpts *opts;
43 QTAILQ_ENTRY(QemuOpt) next;
44};
45
46struct QemuOpts {
47 char *id;
48 QemuOptsList *list;
49 Location loc;
50 QTAILQ_HEAD(QemuOptHead, QemuOpt) head;
51 QTAILQ_ENTRY(QemuOpts) next;
52};
