45enum QemuOptType {
46 QEMU_OPT_STRING = 0, /* no parsing (use string as-is) */
47 QEMU_OPT_BOOL, /* on/off */
48 QEMU_OPT_NUMBER, /* simple number */
49 QEMU_OPT_SIZE, /* size, accepts (K)ilo, (M)ega, (G)iga, (T)era postfix */
50};
51
52typedef struct QemuOptDesc {
53 const char *name;
54 enum QemuOptType type;
55 const char *help;
56 const char *def_value_str;
57} QemuOptDesc;
58
59struct QemuOptsList {
60 const char *name;
61 const char *implied_opt_name;
62 bool merge_lists; /* Merge multiple uses of option into a single list? */
63 QTAILQ_HEAD(, QemuOpts) head;
64 QemuOptDesc desc[];
65};

75 * list of the valid values for a command line option (as defined
76 * by is_help_option()).
