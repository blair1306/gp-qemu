30typedef struct qemu_acl_entry qemu_acl_entry;
31typedef struct qemu_acl qemu_acl;
32
33struct qemu_acl_entry {
34 char *match;
35 int deny;
36
37 QTAILQ_ENTRY(qemu_acl_entry) next;
38};

40struct qemu_acl {
41 char *aclname;
42 unsigned int nentries;
43 QTAILQ_HEAD(,qemu_acl_entry) entries;
44 int defaultDeny;
45};

60 int deny,
61 const char *match,
62 int index);
63int qemu_acl_remove(qemu_acl *acl,
64 const char *match);
