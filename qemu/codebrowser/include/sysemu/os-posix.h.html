56/**
57 * qemu_alloc_stack:
58 * @sz: pointer to a size_t holding the requested usable stack size
59 *
60 * Allocate memory that can be used as a stack, for instance for
61 * coroutines. If the memory cannot be allocated, this function
62 * will abort (like g_malloc()). This function also inserts an
63 * additional guard page to catch a potential stack overflow.
64 * Note that the memory required for the guard page and alignment
65 * and minimal stack size restrictions will increase the value of sz.
66 *
67 * The allocated stack must be freed with qemu_free_stack().
68 *
69 * Returns: pointer to (the lowest address of) the stack memory.
70 */
71void *qemu_alloc_stack(size_t *sz);
72
73/**
74 * qemu_free_stack:
75 * @stack: stack to free
76 * @sz: size of stack in bytes
77 *
78 * Free a stack allocated via qemu_alloc_stack(). Note that sz must
79 * be exactly the adjusted stack size returned by qemu_alloc_stack.
80 */
81void qemu_free_stack(void *stack, size_t sz);
