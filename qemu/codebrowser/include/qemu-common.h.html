2/* Common header file that is included by all of QEMU.
3 *
4 * This file is supposed to be included only by .c files. No header file should
5 * depend on qemu-common.h, as this would easily lead to circular header
6 * dependencies.
7 *
8 * If a header file uses a definition from qemu-common.h, that definition
9 * must be moved to a separate header file, and the header that uses it
10 * must include that header.
11 */
12#ifndef QEMU_COMMON_H
13#define QEMU_COMMON_H

90/**
91 * set_preferred_target_page_bits:
92 * @bits: number of bits needed to represent an address within the page
93 *
94 * Set the preferred target page size (the actual target page
95 * size may be smaller than any given CPU's preference).
96 * Returns true on success, false on failure (which can only happen
97 * if this is called after the system has already finalized its
98 * choice of page size and the requested page size is smaller than that).
99 */

102/**
103 * Sends a (part of) iovec down a socket, yielding when the socket is full, or
104 * Receives data into a (part of) iovec from a socket,
105 * yielding when there is no data in the socket.
106 * The same interface as qemu_sendv_recvv(), with added yielding.
107 * XXX should mark these as coroutine_fn
108 */
109ssize_t qemu_co_sendv_recvv(int sockfd, struct iovec *iov, unsigned iov_cnt,
110 size_t offset, size_t bytes, bool do_send);
