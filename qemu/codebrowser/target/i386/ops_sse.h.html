41void glue(helper_psrlw, SUFFIX)(CPUX86State *env, Reg *d, Reg *s)
42{
43 int shift;
44
45 if (s->Q(0) > 15) {
46 d->Q(0) = 0;
47#if SHIFT == 1
48 d->Q(1) = 0;
49#endif
50 } else {
51 shift = s->B(0);
52 d->W(0) >>= shift;
53 d->W(1) >>= shift;
54 d->W(2) >>= shift;
55 d->W(3) >>= shift;
56#if SHIFT == 1
57 d->W(4) >>= shift;
58 d->W(5) >>= shift;
59 d->W(6) >>= shift;
60 d->W(7) >>= shift;
61#endif
62 }
63}

461 val += abs1(d->B(5) - s->B(5));
