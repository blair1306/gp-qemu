44/*
45 * TCP header.
46 * Per RFC 793, September, 1981.
47 */
48#define tcphdr slirp_tcphdr
49struct tcphdr {
50 uint16_t th_sport; /* source port */
51 uint16_t th_dport; /* destination port */
52 tcp_seq th_seq; /* sequence number */
53 tcp_seq th_ack; /* acknowledgement number */
54#ifdef HOST_WORDS_BIGENDIAN
55 uint8_t th_off:4, /* data offset */
56 th_x2:4; /* (unused) */
57#else
58 uint8_t th_x2:4, /* (unused) */
59 th_off:4; /* data offset */
60#endif
61 uint8_t th_flags;
62 uint16_t th_win; /* window */
63 uint16_t th_sum; /* checksum */
64 uint16_t th_urp; /* urgent pointer */
65};

99/*
100 * Default maximum segment size for TCP.
101 * With an IP MSS of 576, this is 536,
102 * but 512 is probably more convenient.
103 * This should be defined as MIN(512, IP_MSS - sizeof (struct tcpiphdr)).
104 *
105 * We make this 1460 because we only care about Ethernet in the qemu context.
106 */

129/*
130 * TCP FSM state definitions.
131 * Per RFC793, September, 1981.
132 */
133
134#define TCP_NSTATES 11
135
136#define TCPS_CLOSED 0 /* closed */
137#define TCPS_LISTEN 1 /* listening for connection */
138#define TCPS_SYN_SENT 2 /* active, have sent syn */
139#define TCPS_SYN_RECEIVED 3 /* have send and received syn */
140/* states < TCPS_ESTABLISHED are those where connections not established */
141#define TCPS_ESTABLISHED 4 /* established */
142#define TCPS_CLOSE_WAIT 5 /* rcvd fin, waiting for close */
143/* states > TCPS_CLOSE_WAIT are those where user has closed */
144#define TCPS_FIN_WAIT_1 6 /* have closed, sent fin */
145#define TCPS_CLOSING 7 /* closed xchd FIN; await FIN ACK */
146#define TCPS_LAST_ACK 8 /* had fin and close; await FIN ACK */
147/* states > TCPS_CLOSE_WAIT && < TCPS_FIN_WAIT_2 await ACK of FIN */
148#define TCPS_FIN_WAIT_2 9 /* have closed, fin is acked */
149#define TCPS_TIME_WAIT 10 /* in 2*msl quiet wait after close */
150
151#define TCPS_HAVERCVDSYN(s) ((s) >= TCPS_SYN_RECEIVED)
152#define TCPS_HAVEESTABLISHED(s) ((s) >= TCPS_ESTABLISHED)
153#define TCPS_HAVERCVDFIN(s) ((s) >= TCPS_TIME_WAIT)

155/*
156 * TCP sequence numbers are 32 bit integers operated
157 * on with modular arithmetic. These macros can be
158 * used to compare such integers.
159 */
160#define SEQ_LT(a,b) ((int)((a)-(b)) < 0)
161#define SEQ_LEQ(a,b) ((int)((a)-(b)) <= 0)
162#define SEQ_GT(a,b) ((int)((a)-(b)) > 0)
163#define SEQ_GEQ(a,b) ((int)((a)-(b)) >= 0)
164
165/*
166 * Macros to initialize tcp sequence numbers for
167 * send and receive from initial send and receive
168 * sequence numbers.
169 */
170#define tcp_rcvseqinit(tp) \
171 (tp)->rcv_adv = (tp)->rcv_nxt = (tp)->irs + 1
172
173#define tcp_sendseqinit(tp) \
174 (tp)->snd_una = (tp)->snd_nxt = (tp)->snd_max = (tp)->snd_up = (tp)->iss
