21#ifndef QCRYPTO_IVGENPRIV_H
22#define QCRYPTO_IVGENPRIV_H

24#include "crypto/ivgen.h"
25
26typedef struct QCryptoIVGenDriver QCryptoIVGenDriver;
27
28struct QCryptoIVGenDriver {
29 int (*init)(QCryptoIVGen *ivgen,
30 const uint8_t *key, size_t nkey,
31 Error **errp);
32 int (*calculate)(QCryptoIVGen *ivgen,
33 uint64_t sector,
34 uint8_t *iv, size_t niv,
35 Error **errp);
36 void (*cleanup)(QCryptoIVGen *ivgen);
37};
38
39struct QCryptoIVGen {
40 QCryptoIVGenDriver *driver;
41 void *private;
42
43 QCryptoIVGenAlgorithm algorithm;
44 QCryptoCipherAlgorithm cipher;
45 QCryptoHashAlgorithm hash;
46};
