44static int
45qcrypto_block_qcow_init(QCryptoBlock *block,
46 const char *keysecret,
47 Error **errp)
48{
49 char *password;
50 int ret;
51 uint8_t keybuf[16];
52 int len;
53
54 memset(keybuf, 0, 16);
55
56 password = qcrypto_secret_lookup_as_utf8(keysecret, errp);
57 if (!password) {
58 return -1;
59 }
60
61 len = strlen(password);
62 memcpy(keybuf, password, MIN(len, sizeof(keybuf)));
63 g_free(password);
64
65 block->niv = qcrypto_cipher_get_iv_len(QCRYPTO_CIPHER_ALG_AES_128,
66 QCRYPTO_CIPHER_MODE_CBC);
67 block->ivgen = qcrypto_ivgen_new(QCRYPTO_IVGEN_ALG_PLAIN64,
68 0, 0, NULL, 0, errp);
69 if (!block->ivgen) {
70 ret = -ENOTSUP;
71 goto fail;
72 }
73
74 block->cipher = qcrypto_cipher_new(QCRYPTO_CIPHER_ALG_AES_128,
75 QCRYPTO_CIPHER_MODE_CBC,
76 keybuf, G_N_ELEMENTS(keybuf),
77 errp);
78 if (!block->cipher) {
79 ret = -ENOTSUP;
80 goto fail;
81 }
82
83 block->payload_offset = 0;
84
85 return 0;
86
87 fail:
88 qcrypto_cipher_free(block->cipher);
89 qcrypto_ivgen_free(block->ivgen);
90 return ret;
91}

94static int
95qcrypto_block_qcow_open(QCryptoBlock *block,
96 QCryptoBlockOpenOptions *options,
97 const char *optprefix,
98 QCryptoBlockReadFunc readfunc G_GNUC_UNUSED,
99 void *opaque G_GNUC_UNUSED,
100 unsigned int flags,
101 Error **errp)
102{
103 if (flags & QCRYPTO_BLOCK_OPEN_NO_IO) {
104 return 0;
105 } else {
106 if (!options->u.qcow.key_secret) {
107 error_setg(errp,
108 "Parameter '%skey-secret' is required for cipher",
109 optprefix ? optprefix : "");
110 return -1;
111 }
112 return qcrypto_block_qcow_init(block,
113 options->u.qcow.key_secret, errp);
114 }
115}

118static int
119qcrypto_block_qcow_create(QCryptoBlock *block,
120 QCryptoBlockCreateOptions *options,
121 const char *optprefix,
122 QCryptoBlockInitFunc initfunc G_GNUC_UNUSED,
123 QCryptoBlockWriteFunc writefunc G_GNUC_UNUSED,
124 void *opaque G_GNUC_UNUSED,
125 Error **errp)
126{
127 if (!options->u.qcow.key_secret) {
128 error_setg(errp, "Parameter '%skey-secret' is required for cipher",
129 optprefix ? optprefix : "");
130 return -1;
131 }
132 /* QCow2 has no special header, since everything is hardwired */
133 return qcrypto_block_qcow_init(block, options->u.qcow.key_secret, errp);
134}
