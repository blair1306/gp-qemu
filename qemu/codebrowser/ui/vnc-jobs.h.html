42/* Locks */
43static inline int vnc_trylock_display(VncDisplay *vd)
44{
45 return qemu_mutex_trylock(&vd->mutex);
46}
47
48static inline void vnc_lock_display(VncDisplay *vd)
49{
50 qemu_mutex_lock(&vd->mutex);
51}
52
53static inline void vnc_unlock_display(VncDisplay *vd)
54{
55 qemu_mutex_unlock(&vd->mutex);
56}
57
58static inline void vnc_lock_output(VncState *vs)
59{
60 qemu_mutex_lock(&vs->output_mutex);
61}
62
63static inline void vnc_unlock_output(VncState *vs)
64{
65 qemu_mutex_unlock(&vs->output_mutex);
66}
