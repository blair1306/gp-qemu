10PixelFormat qemu_pixelformat_from_pixman(pixman_format_code_t format)
11{
12 PixelFormat pf;
13 uint8_t bpp;
14
15 bpp = pf.bits_per_pixel = PIXMAN_FORMAT_BPP(format);
16 pf.bytes_per_pixel = PIXMAN_FORMAT_BPP(format) / 8;
17 pf.depth = PIXMAN_FORMAT_DEPTH(format);
18
19 pf.abits = PIXMAN_FORMAT_A(format);
20 pf.rbits = PIXMAN_FORMAT_R(format);
21 pf.gbits = PIXMAN_FORMAT_G(format);
22 pf.bbits = PIXMAN_FORMAT_B(format);
23
24 switch (PIXMAN_FORMAT_TYPE(format)) {
25 case PIXMAN_TYPE_ARGB:
26 pf.ashift = pf.bbits + pf.gbits + pf.rbits;
27 pf.rshift = pf.bbits + pf.gbits;
28 pf.gshift = pf.bbits;
29 pf.bshift = 0;
30 break;
31 case PIXMAN_TYPE_ABGR:
32 pf.ashift = pf.rbits + pf.gbits + pf.bbits;
33 pf.bshift = pf.rbits + pf.gbits;
34 pf.gshift = pf.rbits;
35 pf.rshift = 0;
36 break;
37 case PIXMAN_TYPE_BGRA:
38 pf.bshift = bpp - pf.bbits;
39 pf.gshift = bpp - (pf.bbits + pf.gbits);
40 pf.rshift = bpp - (pf.bbits + pf.gbits + pf.rbits);
41 pf.ashift = 0;
42 break;
43 case PIXMAN_TYPE_RGBA:
44 pf.rshift = bpp - pf.rbits;
45 pf.gshift = bpp - (pf.rbits + pf.gbits);
46 pf.bshift = bpp - (pf.rbits + pf.gbits + pf.bbits);
47 pf.ashift = 0;
48 break;
49 default:
50 g_assert_not_reached();
51 break;
52 }
53
54 pf.amax = (1 << pf.abits) - 1;
55 pf.rmax = (1 << pf.rbits) - 1;
56 pf.gmax = (1 << pf.gbits) - 1;
57 pf.bmax = (1 << pf.bbits) - 1;
58 pf.amask = pf.amax << pf.ashift;
59 pf.rmask = pf.rmax << pf.rshift;
60 pf.gmask = pf.gmax << pf.gshift;
61 pf.bmask = pf.bmax << pf.bshift;
62
63 return pf;
64}

66pixman_format_code_t qemu_default_pixman_format(int bpp, bool native_endian)
67{
68 if (native_endian) {
69 switch (bpp) {
70 case 15:
71 return PIXMAN_x1r5g5b5;
72 case 16:
73 return PIXMAN_r5g6b5;
74 case 24:
75 return PIXMAN_r8g8b8;
76 case 32:
77 return PIXMAN_x8r8g8b8;
78 }
79 } else {
80 switch (bpp) {
81 case 24:
82 return PIXMAN_b8g8r8;
83 case 32:
84 return PIXMAN_b8g8r8x8;
85 break;
86 }
87 }
88 return 0;
89}

91int qemu_pixman_get_type(int rshift, int gshift, int bshift)
92{
93 int type = PIXMAN_TYPE_OTHER;
94
95 if (rshift > gshift && gshift > bshift) {
96 if (bshift == 0) {
97 type = PIXMAN_TYPE_ARGB;
98 } else {
99#if PIXMAN_VERSION >= PIXMAN_VERSION_ENCODE(0, 21, 8)
100 type = PIXMAN_TYPE_RGBA;
101#endif
102 }
103 } else if (rshift < gshift && gshift < bshift) {
104 if (rshift == 0) {
105 type = PIXMAN_TYPE_ABGR;
106 } else {
107#if PIXMAN_VERSION >= PIXMAN_VERSION_ENCODE(0, 16, 0)
108 type = PIXMAN_TYPE_BGRA;
109#endif
110 }
111 }
112 return type;
113}

115pixman_format_code_t qemu_pixman_get_format(PixelFormat *pf)
116{
117 pixman_format_code_t format;
118 int type;
119
120 type = qemu_pixman_get_type(pf->rshift, pf->gshift, pf->bshift);
121 format = PIXMAN_FORMAT(pf->bits_per_pixel, type,
122 pf->abits, pf->rbits, pf->gbits, pf->bbits);
123 if (!pixman_format_supported_source(format)) {
124 return 0;
125 }
126 return format;
127}

129/*
130 * Return true for known-good pixman conversions.
131 *
132 * UIs using pixman for format conversion can hook this into
133 * DisplayChangeListenerOps->dpy_gfx_check_format
134 */
135bool qemu_pixman_check_format(DisplayChangeListener *dcl,
136 pixman_format_code_t format)
137{
138 switch (format) {
139 /* 32 bpp */
140 case PIXMAN_x8r8g8b8:
141 case PIXMAN_a8r8g8b8:
142 case PIXMAN_b8g8r8x8:
143 case PIXMAN_b8g8r8a8:
144 /* 24 bpp */
145 case PIXMAN_r8g8b8:
146 case PIXMAN_b8g8r8:
147 /* 16 bpp */
148 case PIXMAN_x1r5g5b5:
149 case PIXMAN_r5g6b5:
150 return true;
151 default:
152 return false;
153 }
154}

198pixman_color_t qemu_pixman_color(PixelFormat *pf, uint32_t color)
199{
200 pixman_color_t c;
201
202 c.red = ((color & pf->rmask) >> pf->rshift) << (16 - pf->rbits);
203 c.green = ((color & pf->gmask) >> pf->gshift) << (16 - pf->gbits);
204 c.blue = ((color & pf->bmask) >> pf->bshift) << (16 - pf->bbits);
205 c.alpha = ((color & pf->amask) >> pf->ashift) << (16 - pf->abits);
206 return c;
207}

209pixman_image_t *qemu_pixman_glyph_from_vgafont(int height, const uint8_t *font,
210 unsigned int ch)
211{
212 pixman_image_t *glyph;
213 uint8_t *data;
214 bool bit;
215 int x, y;
216
217 glyph = pixman_image_create_bits(PIXMAN_a8, 8, height,
218 NULL, 0);
219 data = (uint8_t *)pixman_image_get_data(glyph);
220
221 font += height * ch;
222 for (y = 0; y < height; y++, font++) {
223 for (x = 0; x < 8; x++, data++) {
224 bit = (*font) & (1 << (7-x));
225 *data = bit ? 0xff : 0x00;
226 }
227 }
228 return glyph;
229}

231void qemu_pixman_glyph_render(pixman_image_t *glyph,
232 pixman_image_t *surface,
233 pixman_color_t *fgcol,
234 pixman_color_t *bgcol,
235 int x, int y, int cw, int ch)
236{
237 pixman_image_t *ifg = pixman_image_create_solid_fill(fgcol);
238 pixman_image_t *ibg = pixman_image_create_solid_fill(bgcol);
239
240 pixman_image_composite(PIXMAN_OP_SRC, ibg, NULL, surface,
241 0, 0, 0, 0,
242 cw * x, ch * y,
243 cw, ch);
244 pixman_image_composite(PIXMAN_OP_OVER, ifg, glyph, surface,
245 0, 0, 0, 0,
246 cw * x, ch * y,
247 cw, ch);
248 pixman_image_unref(ifg);
249 pixman_image_unref(ibg);
250}
