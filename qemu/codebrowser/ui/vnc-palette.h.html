29#ifndef VNC_PALETTE_H
30#define VNC_PALETTE_H
31
32#include "qapi/qmp/qlist.h"
33#include "qemu/queue.h"
34
35#define VNC_PALETTE_HASH_SIZE 256
36#define VNC_PALETTE_MAX_SIZE 256
37
38typedef struct VncPaletteEntry {
39 int idx;
40 uint32_t color;
41 QLIST_ENTRY(VncPaletteEntry) next;
42} VncPaletteEntry;
43
44typedef struct VncPalette {
45 VncPaletteEntry pool[VNC_PALETTE_MAX_SIZE];
46 size_t size;
47 size_t max;
48 int bpp;
49 QLIST_HEAD(,VncPaletteEntry) table[VNC_PALETTE_HASH_SIZE];
50} VncPalette;
51
52VncPalette *palette_new(size_t max, int bpp);
53void palette_init(VncPalette *palette, size_t max, int bpp);
54void palette_destroy(VncPalette *palette);
55
56int palette_put(VncPalette *palette, uint32_t color);
57int palette_idx(const VncPalette *palette, uint32_t color);
58size_t palette_size(const VncPalette *palette);
59
60void palette_iter(const VncPalette *palette,
61 void (*iter)(int idx, uint32_t color, void *opaque),
62 void *opaque);
63uint32_t palette_color(const VncPalette *palette, int idx, bool *found);
64size_t palette_fill(const VncPalette *palette,
65 uint32_t colors[VNC_PALETTE_MAX_SIZE]);
66
67#endif /* VNC_PALETTE_H */
