46void qemu_chr_be_event(Chardev *s, int event)
47{
48 CharBackend *be = s->be;
49
50 /* Keep track if the char device is open */
51 switch (event) {
52 case CHR_EVENT_OPENED:
53 s->be_open = 1;
54 break;
55 case CHR_EVENT_CLOSED:
56 s->be_open = 0;
57 break;
58 }
59
60 if (!be || !be->chr_event) {
61 return;
62 }
63
64 be->chr_event(be->opaque, event);
65}

67/* Not reporting errors from writing to logfile, as logs are
68 * defined to be "best effort" only */
69static void qemu_chr_write_log(Chardev *s, const uint8_t *buf, size_t len)
70{
71 size_t done = 0;
72 ssize_t ret;
73
74 if (s->logfd < 0) {
75 return;
76 }
77
78 while (done < len) {
79 retry:
80 ret = write(s->logfd, buf + done, len - done);
81 if (ret == -1 && errno == EAGAIN) {
82 g_usleep(100);
83 goto retry;
84 }
85
86 if (ret <= 0) {
87 return;
88 }
89 done += ret;
90 }
91}

93static int qemu_chr_write_buffer(Chardev *s,
94 const uint8_t *buf, int len,
95 int *offset, bool write_all)
96{
97 ChardevClass *cc = CHARDEV_GET_CLASS(s);
98 int res = 0;
99 *offset = 0;
100
101 qemu_mutex_lock(&s->chr_write_lock);
102 while (*offset < len) {
103 retry:
104 res = cc->chr_write(s, buf + *offset, len - *offset);
105 if (res < 0 && errno == EAGAIN && write_all) {
106 g_usleep(100);
107 goto retry;
108 }
109
110 if (res <= 0) {
111 break;
112 }
113
114 *offset += res;
115 if (!write_all) {
116 break;
117 }
118 }
119 if (*offset > 0) {
120 qemu_chr_write_log(s, buf, *offset);
121 }
122 qemu_mutex_unlock(&s->chr_write_lock);
123
124 return res;
125}

127int qemu_chr_write(Chardev *s, const uint8_t *buf, int len, bool write_all)
128{
129 int offset = 0;
130 int res;
131
132 if (qemu_chr_replay(s) && replay_mode == REPLAY_MODE_PLAY) {
133 replay_char_write_event_load(&res, &offset);
134 assert(offset <= len);
135 qemu_chr_write_buffer(s, buf, offset, &offset, true);
136 return res;
137 }
138
139 res = qemu_chr_write_buffer(s, buf, len, &offset, write_all);
140
141 if (qemu_chr_replay(s) && replay_mode == REPLAY_MODE_RECORD) {
142 replay_char_write_event_save(res, offset);
143 }
144
145 if (res < 0) {
146 return res;
147 }
148 return offset;
149}

189static void qemu_char_open(Chardev *chr, ChardevBackend *backend,
190 bool *be_opened, Error **errp)
191{
192 ChardevClass *cc = CHARDEV_GET_CLASS(chr);
193 /* Any ChardevCommon member would work */
194 ChardevCommon *common = backend ? backend->u.null.data : NULL;
195
196 if (common && common->has_logfile) {
197 int flags = O_WRONLY | O_CREAT;
198 if (common->has_logappend &&
199 common->logappend) {
200 flags |= O_APPEND;
201 } else {
202 flags |= O_TRUNC;
203 }
204 chr->logfd = qemu_open(common->logfile, flags, 0666);
205 if (chr->logfd < 0) {
206 error_setg_errno(errp, errno,
207 "Unable to open logfile %s",
208 common->logfile);
209 return;
210 }
211 }
212
213 if (cc->open) {
214 cc->open(chr, backend, be_opened, errp);
215 }
216}

264/**
265 * Called after processing of default and command-line-specified
266 * chardevs to deliver CHR_EVENT_OPENED events to any FEs attached
267 * to a mux chardev. This is done here to ensure that
268 * output/prompts/banners are only displayed for the FE that has
269 * focus when initial command-line processing/machine init is
270 * completed.
271 *
272 * After this point, any new FE attached to any new or existing
273 * mux will receive CHR_EVENT_OPENED notifications for the BE
274 * immediately.
275 */
276static int open_muxes(Object *child, void *opaque)
277{
278 if (CHARDEV_IS_MUX(child)) {
279 /* send OPENED to all already-attached FEs */
280 mux_chr_send_all_event(CHARDEV(child), CHR_EVENT_OPENED);
281 /* mark mux as OPENED so any new FEs will immediately receive
282 * OPENED event
283 */
284 qemu_chr_be_event(CHARDEV(child), CHR_EVENT_OPENED);
285 }
286
287 return 0;
288}

293 object_child_foreach(get_chardevs_root(), open_muxes, NULL);

321QemuOpts *qemu_chr_parse_compat(const char *label, const char *filename)
322{
323 char host[65], port[33], width[8], height[8];
324 int pos;
325 const char *p;
326 QemuOpts *opts;
327 Error *local_err = NULL;
328
329 opts = qemu_opts_create(qemu_find_opts("chardev"), label, 1, &local_err);
330 if (local_err) {
331 error_report_err(local_err);
332 return NULL;
333 }
334
335 if (strstart(filename, "mon:", &p)) {
336 filename = p;
337 qemu_opt_set(opts, "mux", "on", &error_abort);
338 if (strcmp(filename, "stdio") == 0) {
339 /* Monitor is muxed to stdio: do not exit on Ctrl+C by default
340 * but pass it to the guest. Handle this only for compat syntax,
341 * for -chardev syntax we have special option for this.
342 * This is what -nographic did, redirecting+muxing serial+monitor
343 * to stdio causing Ctrl+C to be passed to guest. */
344 qemu_opt_set(opts, "signal", "off", &error_abort);
345 }
346 }
347
348 if (strcmp(filename, "null") == 0 ||
349 strcmp(filename, "pty") == 0 ||
350 strcmp(filename, "msmouse") == 0 ||
351 strcmp(filename, "wctablet") == 0 ||
352 strcmp(filename, "braille") == 0 ||
353 strcmp(filename, "testdev") == 0 ||
354 strcmp(filename, "stdio") == 0) {
355 qemu_opt_set(opts, "backend", filename, &error_abort);
356 return opts;
357 }
358 if (strstart(filename, "vc", &p)) {
359 qemu_opt_set(opts, "backend", "vc", &error_abort);
360 if (*p == ':') {
361 if (sscanf(p+1, "%7[0-9]x%7[0-9]", width, height) == 2) {
362 /* pixels */
363 qemu_opt_set(opts, "width", width, &error_abort);
364 qemu_opt_set(opts, "height", height, &error_abort);
365 } else if (sscanf(p+1, "%7[0-9]Cx%7[0-9]C", width, height) == 2) {
366 /* chars */
367 qemu_opt_set(opts, "cols", width, &error_abort);
368 qemu_opt_set(opts, "rows", height, &error_abort);
369 } else {
370 goto fail;
371 }
372 }
373 return opts;
374 }
375 if (strcmp(filename, "con:") == 0) {
376 qemu_opt_set(opts, "backend", "console", &error_abort);
377 return opts;
378 }
379 if (strstart(filename, "COM", NULL)) {
380 qemu_opt_set(opts, "backend", "serial", &error_abort);
381 qemu_opt_set(opts, "path", filename, &error_abort);
382 return opts;
383 }
384 if (strstart(filename, "file:", &p)) {
385 qemu_opt_set(opts, "backend", "file", &error_abort);
386 qemu_opt_set(opts, "path", p, &error_abort);
387 return opts;
388 }
389 if (strstart(filename, "pipe:", &p)) {
390 qemu_opt_set(opts, "backend", "pipe", &error_abort);
391 qemu_opt_set(opts, "path", p, &error_abort);
392 return opts;
393 }
394 if (strstart(filename, "tcp:", &p) ||
395 strstart(filename, "telnet:", &p) ||
396 strstart(filename, "tn3270:", &p)) {
397 if (sscanf(p, "%64[^:]:%32[^,]%n", host, port, &pos) < 2) {
398 host[0] = 0;
399 if (sscanf(p, ":%32[^,]%n", port, &pos) < 1)
400 goto fail;
401 }
402 qemu_opt_set(opts, "backend", "socket", &error_abort);
403 qemu_opt_set(opts, "host", host, &error_abort);
404 qemu_opt_set(opts, "port", port, &error_abort);
405 if (p[pos] == ',') {
406 qemu_opts_do_parse(opts, p+pos+1, NULL, &local_err);
407 if (local_err) {
408 error_report_err(local_err);
409 goto fail;
410 }
411 }
412 if (strstart(filename, "telnet:", &p)) {
413 qemu_opt_set(opts, "telnet", "on", &error_abort);
414 } else if (strstart(filename, "tn3270:", &p)) {
415 qemu_opt_set(opts, "tn3270", "on", &error_abort);
416 }
417 return opts;
418 }
419 if (strstart(filename, "udp:", &p)) {
420 qemu_opt_set(opts, "backend", "udp", &error_abort);
421 if (sscanf(p, "%64[^:]:%32[^@,]%n", host, port, &pos) < 2) {
422 host[0] = 0;
423 if (sscanf(p, ":%32[^@,]%n", port, &pos) < 1) {
424 goto fail;
425 }
426 }
427 qemu_opt_set(opts, "host", host, &error_abort);
428 qemu_opt_set(opts, "port", port, &error_abort);
429 if (p[pos] == '@') {
430 p += pos + 1;
431 if (sscanf(p, "%64[^:]:%32[^,]%n", host, port, &pos) < 2) {
432 host[0] = 0;
433 if (sscanf(p, ":%32[^,]%n", port, &pos) < 1) {
434 goto fail;
435 }
436 }
437 qemu_opt_set(opts, "localaddr", host, &error_abort);
438 qemu_opt_set(opts, "localport", port, &error_abort);
439 }
440 return opts;
441 }
442 if (strstart(filename, "unix:", &p)) {
443 qemu_opt_set(opts, "backend", "socket", &error_abort);
444 qemu_opts_do_parse(opts, p, "path", &local_err);
445 if (local_err) {
446 error_report_err(local_err);
447 goto fail;
448 }
449 return opts;
450 }
451 if (strstart(filename, "/dev/parport", NULL) ||
452 strstart(filename, "/dev/ppi", NULL)) {
453 qemu_opt_set(opts, "backend", "parallel", &error_abort);
454 qemu_opt_set(opts, "path", filename, &error_abort);
455 return opts;
456 }
457 if (strstart(filename, "/dev/", NULL)) {
458 qemu_opt_set(opts, "backend", "serial", &error_abort);
459 qemu_opt_set(opts, "path", filename, &error_abort);
460 return opts;
461 }
462
463fail:
464 qemu_opts_del(opts);
465 return NULL;
466}

479static const ChardevClass *char_get_class(const char *driver, Error **errp)
480{
481 ObjectClass *oc;
482 const ChardevClass *cc;
483 char *typename = g_strdup_printf("chardev-%s", driver);
484
485 oc = object_class_by_name(typename);
486 g_free(typename);
487
488 if (!object_class_dynamic_cast(oc, TYPE_CHARDEV)) {
489 error_setg(errp, "'%s' is not a valid char driver name", driver);
490 return NULL;
491 }
492
493 if (object_class_is_abstract(oc)) {
494 error_setg(errp, QERR_INVALID_PARAMETER_VALUE, "driver",
495 "abstract device type");
496 return NULL;
497 }
498
499 cc = CHARDEV_CLASS(oc);
500 if (cc->internal) {
501 error_setg(errp, "'%s' is not a valid char driver name", driver);
502 return NULL;
503 }
504
505 return cc;
506}

520typedef struct ChadevClassFE {
521 void (*fn)(const char *name, void *opaque);
522 void *opaque;
523} ChadevClassFE;
524
525static void
526chardev_class_foreach(ObjectClass *klass, void *opaque)
527{
528 ChadevClassFE *fe = opaque;
529
530 assert(g_str_has_prefix(object_class_get_name(klass), "chardev-"));
531 if (CHARDEV_CLASS(klass)->internal) {
532 return;
533 }
534
535 fe->fn(object_class_get_name(klass) + 8, fe->opaque);
536}

538static void
539chardev_name_foreach(void (*fn)(const char *name, void *opaque), void *opaque)
540{
541 ChadevClassFE fe = { .fn = fn, .opaque = opaque };
542 int i;
543
544 object_class_foreach(chardev_class_foreach, TYPE_CHARDEV, false, &fe);
545
546 for (i = 0; i < (int)ARRAY_SIZE(chardev_alias_table); i++) {
547 fn(chardev_alias_table[i].alias, opaque);
548 }
549}

559static const char *chardev_alias_translate(const char *name)
560{
561 int i;
562 for (i = 0; i < (int)ARRAY_SIZE(chardev_alias_table); i++) {
563 if (g_strcmp0(chardev_alias_table[i].alias, name) == 0) {
564 return chardev_alias_table[i].typename;
565 }
566 }
567 return name;
568}

570ChardevBackend *qemu_chr_parse_opts(QemuOpts *opts, Error **errp)
571{
572 Error *local_err = NULL;
573 const ChardevClass *cc;
574 ChardevBackend *backend = NULL;
575 const char *name = chardev_alias_translate(qemu_opt_get(opts, "backend"));
576
577 if (name == NULL) {
578 error_setg(errp, "chardev: \"%s\" missing backend",
579 qemu_opts_id(opts));
580 return NULL;
581 }
582
583 cc = char_get_class(name, errp);
584 if (cc == NULL) {
585 return NULL;
586 }
587
588 backend = g_new0(ChardevBackend, 1);
589 backend->type = CHARDEV_BACKEND_KIND_NULL;
590
591 if (cc->parse) {
592 cc->parse(opts, backend, &local_err);
593 if (local_err) {
594 error_propagate(errp, local_err);
595 qapi_free_ChardevBackend(backend);
596 return NULL;
597 }
598 } else {
599 ChardevCommon *ccom = g_new0(ChardevCommon, 1);
600 qemu_chr_parse_common(opts, ccom);
601 backend->u.null.data = ccom; /* Any ChardevCommon member would work */
602 }
603
604 return backend;
605}

607Chardev *qemu_chr_new_from_opts(QemuOpts *opts, Error **errp)
608{
609 const ChardevClass *cc;
610 Chardev *chr = NULL;
611 ChardevBackend *backend = NULL;
612 const char *name = chardev_alias_translate(qemu_opt_get(opts, "backend"));
613 const char *id = qemu_opts_id(opts);
614 char *bid = NULL;
615
616 if (name && is_help_option(name)) {
617 GString *str = g_string_new("");
618
619 chardev_name_foreach(help_string_append, str);
620
621 error_report("Available chardev backend types: %s", str->str);
622 g_string_free(str, true);
623 exit(0);
624 }
625
626 if (id == NULL) {
627 error_setg(errp, "chardev: no id specified");
628 return NULL;
629 }
630
631 backend = qemu_chr_parse_opts(opts, errp);
632 if (backend == NULL) {
633 return NULL;
634 }
635
636 cc = char_get_class(name, errp);
637 if (cc == NULL) {
638 goto out;
639 }
640
641 if (qemu_opt_get_bool(opts, "mux", 0)) {
642 bid = g_strdup_printf("%s-base", id);
643 }
644
645 chr = qemu_chardev_new(bid ? bid : id,
646 object_class_get_name(OBJECT_CLASS(cc)),
647 backend, errp);
648
649 if (chr == NULL) {
650 goto out;
651 }
652
653 if (bid) {
654 Chardev *mux;
655 qapi_free_ChardevBackend(backend);
656 backend = g_new0(ChardevBackend, 1);
657 backend->type = CHARDEV_BACKEND_KIND_MUX;
658 backend->u.mux.data = g_new0(ChardevMux, 1);
659 backend->u.mux.data->chardev = g_strdup(bid);
660 mux = qemu_chardev_new(id, TYPE_CHARDEV_MUX, backend, errp);
661 if (mux == NULL) {
662 object_unparent(OBJECT(chr));
663 chr = NULL;
664 goto out;
665 }
666 chr = mux;
667 }
668
669out:
670 qapi_free_ChardevBackend(backend);
671 g_free(bid);
672 return chr;
673}

675Chardev *qemu_chr_new_noreplay(const char *label, const char *filename)
676{
677 const char *p;
678 Chardev *chr;
679 QemuOpts *opts;
680 Error *err = NULL;
681
682 if (strstart(filename, "chardev:", &p)) {
683 return qemu_chr_find(p);
684 }
685
686 opts = qemu_chr_parse_compat(label, filename);
687 if (!opts)
688 return NULL;
689
690 chr = qemu_chr_new_from_opts(opts, &err);
691 if (err) {
692 error_report_err(err);
693 }
694 if (chr && qemu_opt_get_bool(opts, "mux", 0)) {
695 monitor_init(chr, MONITOR_USE_READLINE);
696 }
697 qemu_opts_del(opts);
698 return chr;
699}

701Chardev *qemu_chr_new(const char *label, const char *filename)
702{
703 Chardev *chr;
704 chr = qemu_chr_new_noreplay(label, filename);
705 if (chr) {
706 if (replay_mode != REPLAY_MODE_NONE) {
707 qemu_chr_set_feature(chr, QEMU_CHAR_FEATURE_REPLAY);
708 }
709 if (qemu_chr_replay(chr) && CHARDEV_GET_CLASS(chr)->chr_ioctl) {
710 error_report("Replay: ioctl is not supported "
711 "for serial devices yet");
712 }
713 replay_register_char_driver(chr);
714 }
715 return chr;
716}

718static int qmp_query_chardev_foreach(Object *obj, void *data)
719{
720 Chardev *chr = CHARDEV(obj);
721 ChardevInfoList **list = data;
722 ChardevInfoList *info = g_malloc0(sizeof(*info));
723
724 info->value = g_malloc0(sizeof(*info->value));
725 info->value->label = g_strdup(chr->label);
726 info->value->filename = g_strdup(chr->filename);
727 info->value->frontend_open = chr->be && chr->be->fe_open;

882Chardev *qemu_chardev_new(const char *id, const char *typename,
883 ChardevBackend *backend,
884 Error **errp)
885{
886 Object *obj;
887 Chardev *chr = NULL;
888 Error *local_err = NULL;
889 bool be_opened = true;
890
891 assert(g_str_has_prefix(typename, "chardev-"));
892
893 obj = object_new(typename);
894 chr = CHARDEV(obj);
895 chr->label = g_strdup(id);
896
897 qemu_char_open(chr, backend, &be_opened, &local_err);
898 if (local_err) {
899 goto end;
900 }
901
902 if (!chr->filename) {
903 chr->filename = g_strdup(typename + 8);
904 }
905 if (be_opened) {
906 qemu_chr_be_event(chr, CHR_EVENT_OPENED);
907 }
908
909 if (id) {
910 object_property_add_child(get_chardevs_root(), id, obj, &local_err);
911 if (local_err) {
912 goto end;
913 }
914 object_unref(obj);
915 }
916
917end:
918 if (local_err) {
919 error_propagate(errp, local_err);
920 object_unref(obj);
921 return NULL;
922 }
923
924 return chr;
925}

927ChardevReturn *qmp_chardev_add(const char *id, ChardevBackend *backend,
928 Error **errp)
929{
930 const ChardevClass *cc;
931 ChardevReturn *ret;
932 Chardev *chr;
933
934 cc = char_get_class(ChardevBackendKind_lookup[backend->type], errp);
935 if (!cc) {
936 return NULL;
937 }
938
939 chr = qemu_chardev_new(id, object_class_get_name(OBJECT_CLASS(cc)),
940 backend, errp);
941 if (!chr) {
942 return NULL;
943 }
944
945 ret = g_new0(ChardevReturn, 1);
946 if (CHARDEV_IS_PTY(chr)) {
947 ret->pty = g_strdup(chr->filename + 4);
948 ret->has_pty = true;
949 }
950
951 return ret;
952}
953
954ChardevReturn *qmp_chardev_change(const char *id, ChardevBackend *backend,
955 Error **errp)
956{
957 CharBackend *be;
958 const ChardevClass *cc;
959 Chardev *chr, *chr_new;
960 bool closed_sent = false;
961 ChardevReturn *ret;
962
963 chr = qemu_chr_find(id);
964 if (!chr) {
965 error_setg(errp, "Chardev '%s' does not exist", id);
966 return NULL;
967 }
968
969 if (CHARDEV_IS_MUX(chr)) {
970 error_setg(errp, "Mux device hotswap not supported yet");
971 return NULL;
972 }
973
974 if (qemu_chr_replay(chr)) {
975 error_setg(errp,
976 "Chardev '%s' cannot be changed in record/replay mode", id);
977 return NULL;
978 }
979
980 be = chr->be;
981 if (!be) {
982 /* easy case */
983 object_unparent(OBJECT(chr));
984 return qmp_chardev_add(id, backend, errp);
985 }
986
987 if (!be->chr_be_change) {
988 error_setg(errp, "Chardev user does not support chardev hotswap");
989 return NULL;
990 }
991
992 cc = char_get_class(ChardevBackendKind_lookup[backend->type], errp);
993 if (!cc) {
994 return NULL;
995 }
996
997 chr_new = qemu_chardev_new(NULL, object_class_get_name(OBJECT_CLASS(cc)),
998 backend, errp);
999 if (!chr_new) {
1000 return NULL;
1001 }
1002 chr_new->label = g_strdup(id);
1003
1004 if (chr->be_open && !chr_new->be_open) {
1005 qemu_chr_be_event(chr, CHR_EVENT_CLOSED);
1006 closed_sent = true;
1007 }
1008
1009 chr->be = NULL;
1010 qemu_chr_fe_init(be, chr_new, &error_abort);
1011
1012 if (be->chr_be_change(be->opaque) < 0) {
1013 error_setg(errp, "Chardev '%s' change failed", chr_new->label);
1014 chr_new->be = NULL;
1015 qemu_chr_fe_init(be, chr, &error_abort);
1016 if (closed_sent) {
1017 qemu_chr_be_event(chr, CHR_EVENT_OPENED);
1018 }
1019 object_unref(OBJECT(chr_new));
1020 return NULL;
1021 }
1022
1023 object_unparent(OBJECT(chr));
1024 object_property_add_child(get_chardevs_root(), chr_new->label,
1025 OBJECT(chr_new), &error_abort);
1026 object_unref(OBJECT(chr_new));
1027
1028 ret = g_new0(ChardevReturn, 1);
1029 if (CHARDEV_IS_PTY(chr_new)) {
1030 ret->pty = g_strdup(chr_new->filename + 4);
1031 ret->has_pty = true;
1032 }
1033
1034 return ret;
1035}

1037void qmp_chardev_remove(const char *id, Error **errp)
1038{
1039 Chardev *chr;
1040
1041 chr = qemu_chr_find(id);
1042 if (chr == NULL) {
1043 error_setg(errp, "Chardev '%s' not found", id);
1044 return;
1045 }
1046 if (qemu_chr_is_busy(chr)) {
1047 error_setg(errp, "Chardev '%s' is busy", id);
1048 return;
1049 }
1050 if (qemu_chr_replay(chr)) {
1051 error_setg(errp,
1052 "Chardev '%s' cannot be unplugged in record/replay mode", id);
1053 return;
1054 }
1055 object_unparent(OBJECT(chr));
1056}
