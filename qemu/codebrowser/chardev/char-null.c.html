24#include "qemu/osdep.h"
25#include "chardev/char.h"
26
27static void null_chr_open(Chardev *chr,
28 ChardevBackend *backend,
29 bool *be_opened,
30 Error **errp)
31{
32 *be_opened = false;
33}
34
35static void char_null_class_init(ObjectClass *oc, void *data)
36{
37 ChardevClass *cc = CHARDEV_CLASS(oc);
38
39 cc->open = null_chr_open;
40}
41
42static const TypeInfo char_null_type_info = {
43 .name = TYPE_CHARDEV_NULL,
44 .parent = TYPE_CHARDEV,
45 .instance_size = sizeof(Chardev),
46 .class_init = char_null_class_init,
47};
48
49static void register_types(void)
50{
51 type_register_static(&char_null_type_info);
52}
53
54type_init(register_types);
