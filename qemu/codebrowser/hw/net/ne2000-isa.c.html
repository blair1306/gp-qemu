35#define TYPE_ISA_NE2000 "ne2k_isa"
36#define ISA_NE2000(obj) OBJECT_CHECK(ISANE2000State, (obj), TYPE_ISA_NE2000)

38typedef struct ISANE2000State {
39 ISADevice parent_obj;
40
41 uint32_t iobase;
42 uint32_t isairq;
43 NE2000State ne2000;
44} ISANE2000State;

46static NetClientInfo net_ne2000_isa_info = {
47 .type = NET_CLIENT_DRIVER_NIC,
48 .size = sizeof(NICState),
49 .receive = ne2000_receive,
50};

52static const VMStateDescription vmstate_isa_ne2000 = {
53 .name = "ne2000",
54 .version_id = 2,
55 .minimum_version_id = 0,
56 .fields = (VMStateField[]) {
57 VMSTATE_STRUCT(ne2000, ISANE2000State, 0, vmstate_ne2000, NE2000State),
58 VMSTATE_END_OF_LIST()
59 }
60};

62static void isa_ne2000_realizefn(DeviceState *dev, Error **errp)
63{
64 ISADevice *isadev = ISA_DEVICE(dev);
65 ISANE2000State *isa = ISA_NE2000(dev);
66 NE2000State *s = &isa->ne2000;
67
68 ne2000_setup_io(s, DEVICE(isadev), 0x20);
69 isa_register_ioport(isadev, &s->io, isa->iobase);
70
71 isa_init_irq(isadev, &s->irq, isa->isairq);
72
73 qemu_macaddr_default_if_unset(&s->c.macaddr);
74 ne2000_reset(s);
75
76 s->nic = qemu_new_nic(&net_ne2000_isa_info, &s->c,
77 object_get_typename(OBJECT(dev)), dev->id, s);
78 qemu_format_nic_info_str(qemu_get_queue(s->nic), s->c.macaddr.a);
79}

81static Property ne2000_isa_properties[] = {
82 DEFINE_PROP_UINT32("iobase", ISANE2000State, iobase, 0x300),
83 DEFINE_PROP_UINT32("irq", ISANE2000State, isairq, 9),
84 DEFINE_NIC_PROPERTIES(ISANE2000State, ne2000.c),
85 DEFINE_PROP_END_OF_LIST(),
86};

88static void isa_ne2000_class_initfn(ObjectClass *klass, void *data)
89{
90 DeviceClass *dc = DEVICE_CLASS(klass);
91
92 dc->realize = isa_ne2000_realizefn;
93 dc->props = ne2000_isa_properties;
94 dc->vmsd = &vmstate_isa_ne2000;
95 set_bit(DEVICE_CATEGORY_NETWORK, dc->categories);
96}

98static void isa_ne2000_get_bootindex(Object *obj, Visitor *v,
99 const char *name, void *opaque,
100 Error **errp)
101{
102 ISANE2000State *isa = ISA_NE2000(obj);
103 NE2000State *s = &isa->ne2000;
104
105 visit_type_int32(v, name, &s->c.bootindex, errp);
106}
107
108static void isa_ne2000_set_bootindex(Object *obj, Visitor *v,
109 const char *name, void *opaque,
110 Error **errp)
111{
112 ISANE2000State *isa = ISA_NE2000(obj);
113 NE2000State *s = &isa->ne2000;
114 int32_t boot_index;
115 Error *local_err = NULL;
116
117 visit_type_int32(v, name, &boot_index, &local_err);
118 if (local_err) {
119 goto out;
120 }
121 /* check whether bootindex is present in fw_boot_order list */
122 check_boot_index(boot_index, &local_err);
123 if (local_err) {
124 goto out;
125 }
126 /* change bootindex to a new one */
127 s->c.bootindex = boot_index;
128
129out:
130 error_propagate(errp, local_err);
131}
132
133static void isa_ne2000_instance_init(Object *obj)
134{
135 object_property_add(obj, "bootindex", "int32",
136 isa_ne2000_get_bootindex,
137 isa_ne2000_set_bootindex, NULL, NULL, NULL);
138 object_property_set_int(obj, -1, "bootindex", NULL);
139}
