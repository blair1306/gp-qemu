35void init_pam(DeviceState *dev, MemoryRegion *ram_memory,
36 MemoryRegion *system_memory, MemoryRegion *pci_address_space,
37 PAMMemoryRegion *mem, uint32_t start, uint32_t size)
38{
39 int i;
40
41 /* RAM */
42 memory_region_init_alias(&mem->alias[3], OBJECT(dev), "pam-ram", ram_memory,
43 start, size);
44 /* ROM (XXX: not quite correct) */
45 memory_region_init_alias(&mem->alias[1], OBJECT(dev), "pam-rom", ram_memory,
46 start, size);
47 memory_region_set_readonly(&mem->alias[1], true);
48
49 /* XXX: should distinguish read/write cases */
50 memory_region_init_alias(&mem->alias[0], OBJECT(dev), "pam-pci", pci_address_space,
51 start, size);
52 memory_region_init_alias(&mem->alias[2], OBJECT(dev), "pam-pci", ram_memory,
53 start, size);
54
55 for (i = 0; i < 4; ++i) {
56 memory_region_set_enabled(&mem->alias[i], false);
57 memory_region_add_subregion_overlap(system_memory, start,
58 &mem->alias[i], 1);
59 }
60 mem->current = 0;
61}
62
63void pam_update(PAMMemoryRegion *pam, int idx, uint8_t val)
64{
65 assert(0 <= idx && idx <= 12);
66
67 memory_region_set_enabled(&pam->alias[pam->current], false);
68 pam->current = (val >> ((!(idx & 1)) * 4)) & PAM_ATTR_MASK;
69 memory_region_set_enabled(&pam->alias[pam->current], true);
70}
