49static void idebus_unrealize(BusState *bus, Error **errp)
50{
51 IDEBus *ibus = IDE_BUS(bus);
52
53 if (ibus->vmstate) {
54 qemu_del_vm_change_state_handler(ibus->vmstate);
55 }
56}

65void ide_bus_new(IDEBus *idebus, size_t idebus_size, DeviceState *dev,
66 int bus_id, int max_units)
67{
68 qbus_create_inplace(idebus, idebus_size, TYPE_IDE_BUS, dev, NULL);
69 idebus->bus_id = bus_id;
70 idebus->max_units = max_units;
71}

73static char *idebus_get_fw_dev_path(DeviceState *dev)
74{
75 char path[30];
76
77 snprintf(path, sizeof(path), "%s@%x", qdev_fw_name(dev),
78 ((IDEBus*)dev->parent_bus)->bus_id);
79
80 return g_strdup(path);
81}

83static int ide_qdev_init(DeviceState *qdev)
84{
85 IDEDevice *dev = IDE_DEVICE(qdev);
86 IDEDeviceClass *dc = IDE_DEVICE_GET_CLASS(dev);
87 IDEBus *bus = DO_UPCAST(IDEBus, qbus, qdev->parent_bus);
88
89 if (dev->unit == -1) {
90 dev->unit = bus->master ? 1 : 0;
91 }
92
93 if (dev->unit >= bus->max_units) {
94 error_report("Can't create IDE unit %d, bus supports only %d units",
95 dev->unit, bus->max_units);
96 goto err;
97 }
98
99 switch (dev->unit) {
100 case 0:
101 if (bus->master) {
102 error_report("IDE unit %d is in use", dev->unit);
103 goto err;
104 }
105 bus->master = dev;
106 break;
107 case 1:
108 if (bus->slave) {
109 error_report("IDE unit %d is in use", dev->unit);
110 goto err;
111 }
112 bus->slave = dev;
113 break;
114 default:
115 error_report("Invalid IDE unit %d", dev->unit);
116 goto err;
117 }
118 return dc->init(dev);
119
120err:
121 return -1;
122}

124IDEDevice *ide_create_drive(IDEBus *bus, int unit, DriveInfo *drive)
125{
126 DeviceState *dev;
127
128 dev = qdev_create(&bus->qbus, drive->media_cd ? "ide-cd" : "ide-hd");
129 qdev_prop_set_uint32(dev, "unit", unit);
130 qdev_prop_set_drive(dev, "drive", blk_by_legacy_dinfo(drive),
131 &error_fatal);
132 qdev_init_nofail(dev);
133 return DO_UPCAST(IDEDevice, qdev, dev);
134}

136int ide_get_geometry(BusState *bus, int unit,
137 int16_t *cyls, int8_t *heads, int8_t *secs)
138{
139 IDEState *s = &DO_UPCAST(IDEBus, qbus, bus)->ifs[unit];
140
141 if (s->drive_kind != IDE_HD || !s->blk) {
142 return -1;
143 }
144
145 *cyls = s->cylinders;
146 *heads = s->heads;
147 *secs = s->sectors;
148 return 0;
149}

151int ide_get_bios_chs_trans(BusState *bus, int unit)
152{
153 return DO_UPCAST(IDEBus, qbus, bus)->ifs[unit].chs_trans;
154}

162static int ide_dev_initfn(IDEDevice *dev, IDEDriveKind kind)
163{
164 IDEBus *bus = DO_UPCAST(IDEBus, qbus, dev->qdev.parent_bus);
165 IDEState *s = bus->ifs + dev->unit;
166 Error *err = NULL;
167 int ret;
168
169 if (!dev->conf.blk) {
170 if (kind != IDE_CD) {
171 error_report("No drive specified");
172 return -1;
173 } else {
174 /* Anonymous BlockBackend for an empty drive */
175 dev->conf.blk = blk_new(0, BLK_PERM_ALL);
176 ret = blk_attach_dev(dev->conf.blk, &dev->qdev);
177 assert(ret == 0);
178 }
179 }
180
181 if (dev->conf.discard_granularity == -1) {
182 dev->conf.discard_granularity = 512;
183 } else if (dev->conf.discard_granularity &&
184 dev->conf.discard_granularity != 512) {
185 error_report("discard_granularity must be 512 for ide");
186 return -1;
187 }
188
189 blkconf_blocksizes(&dev->conf);
190 if (dev->conf.logical_block_size != 512) {
191 error_report("logical_block_size must be 512 for IDE");
192 return -1;
193 }
194
195 blkconf_serial(&dev->conf, &dev->serial);
196 if (kind != IDE_CD) {
197 blkconf_geometry(&dev->conf, &dev->chs_trans, 65535, 16, 255, &err);
198 if (err) {
199 error_report_err(err);
200 return -1;
201 }
202 }
203 blkconf_apply_backend_options(&dev->conf, kind == IDE_CD, kind != IDE_CD,
204 &err);
205 if (err) {
206 error_report_err(err);
207 return -1;
208 }
209
210 if (ide_init_drive(s, dev->conf.blk, kind,
211 dev->version, dev->serial, dev->model, dev->wwn,
212 dev->conf.cyls, dev->conf.heads, dev->conf.secs,
213 dev->chs_trans) < 0) {
214 return -1;
215 }
216
217 if (!dev->version) {
218 dev->version = g_strdup(s->version);
219 }
220 if (!dev->serial) {
221 dev->serial = g_strdup(s->drive_serial_str);
222 }
223
224 add_boot_device_path(dev->conf.bootindex, &dev->qdev,
225 dev->unit ? "/disk@1" : "/disk@0");
226
227 return 0;
228}

230static void ide_dev_get_bootindex(Object *obj, Visitor *v, const char *name,
231 void *opaque, Error **errp)
232{
233 IDEDevice *d = IDE_DEVICE(obj);
234
235 visit_type_int32(v, name, &d->conf.bootindex, errp);
236}
237
238static void ide_dev_set_bootindex(Object *obj, Visitor *v, const char *name,
239 void *opaque, Error **errp)
240{
241 IDEDevice *d = IDE_DEVICE(obj);
242 int32_t boot_index;
243 Error *local_err = NULL;
244
245 visit_type_int32(v, name, &boot_index, &local_err);
246 if (local_err) {
247 goto out;
248 }
249 /* check whether bootindex is present in fw_boot_order list */
250 check_boot_index(boot_index, &local_err);
251 if (local_err) {
252 goto out;
253 }
254 /* change bootindex to a new one */
255 d->conf.bootindex = boot_index;
256
257 if (d->unit != -1) {
258 add_boot_device_path(d->conf.bootindex, &d->qdev,
259 d->unit ? "/disk@1" : "/disk@0");
260 }
261out:
262 error_propagate(errp, local_err);
263}

265static void ide_dev_instance_init(Object *obj)
266{
267 object_property_add(obj, "bootindex", "int32",
268 ide_dev_get_bootindex,
269 ide_dev_set_bootindex, NULL, NULL, NULL);
270 object_property_set_int(obj, -1, "bootindex", NULL);
271}

283static int ide_drive_initfn(IDEDevice *dev)
284{
285 DriveInfo *dinfo = NULL;
286
287 if (dev->conf.blk) {
288 dinfo = blk_legacy_dinfo(dev->conf.blk);
289 }
290
291 return ide_dev_initfn(dev, dinfo && dinfo->media_cd ? IDE_CD : IDE_HD);
292}
