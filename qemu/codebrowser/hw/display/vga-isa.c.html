35#define TYPE_ISA_VGA "isa-vga"
36#define ISA_VGA(obj) OBJECT_CHECK(ISAVGAState, (obj), TYPE_ISA_VGA)
37
38typedef struct ISAVGAState {
39 ISADevice parent_obj;
40
41 struct VGACommonState state;
42 PortioList portio_vga;
43 PortioList portio_vbe;
44} ISAVGAState;
45
46static void vga_isa_reset(DeviceState *dev)
47{
48 ISAVGAState *d = ISA_VGA(dev);
49 VGACommonState *s = &d->state;
50
51 vga_common_reset(s);
52}
53
54static void vga_isa_realizefn(DeviceState *dev, Error **errp)
55{
56 ISADevice *isadev = ISA_DEVICE(dev);
57 ISAVGAState *d = ISA_VGA(dev);
58 VGACommonState *s = &d->state;
59 MemoryRegion *vga_io_memory;
60 const MemoryRegionPortio *vga_ports, *vbe_ports;
61
62 vga_common_init(s, OBJECT(dev), true);
63 s->legacy_address_space = isa_address_space(isadev);
64 vga_io_memory = vga_init_io(s, OBJECT(dev), &vga_ports, &vbe_ports);
65 isa_register_portio_list(isadev, &d->portio_vga,
66 0x3b0, vga_ports, s, "vga");
67 if (vbe_ports) {
68 isa_register_portio_list(isadev, &d->portio_vbe,
69 0x1ce, vbe_ports, s, "vbe");
70 }
71 memory_region_add_subregion_overlap(isa_address_space(isadev),
72 0x000a0000,
73 vga_io_memory, 1);
74 memory_region_set_coalescing(vga_io_memory);
75 s->con = graphic_console_init(DEVICE(dev), 0, s->hw_ops, s);
76
77 vga_init_vbe(s, OBJECT(dev), isa_address_space(isadev));
78 /* ROM BIOS */
79 rom_add_vga(VGABIOS_FILENAME);
80}
81
82static Property vga_isa_properties[] = {
83 DEFINE_PROP_UINT32("vgamem_mb", ISAVGAState, state.vram_size_mb, 8),
84 DEFINE_PROP_END_OF_LIST(),
85};
86
87static void vga_isa_class_initfn(ObjectClass *klass, void *data)
88{
89 DeviceClass *dc = DEVICE_CLASS(klass);
90
91 dc->realize = vga_isa_realizefn;
92 dc->reset = vga_isa_reset;
93 dc->vmsd = &vmstate_vga_common;
94 dc->props = vga_isa_properties;
95 set_bit(DEVICE_CATEGORY_DISPLAY, dc->categories);
96}
97
98static const TypeInfo vga_isa_info = {
99 .name = TYPE_ISA_VGA,
100 .parent = TYPE_ISA_DEVICE,
101 .instance_size = sizeof(ISAVGAState),
102 .class_init = vga_isa_class_initfn,
103};
104
105static void vga_isa_register_types(void)
106{
107 type_register_static(&vga_isa_info);
108}
