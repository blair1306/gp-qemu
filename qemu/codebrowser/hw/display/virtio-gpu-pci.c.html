20static Property virtio_gpu_pci_properties[] = {
21 DEFINE_VIRTIO_GPU_PCI_PROPERTIES(VirtIOPCIProxy),
22 DEFINE_PROP_END_OF_LIST(),
23};
24
25static void virtio_gpu_pci_realize(VirtIOPCIProxy *vpci_dev, Error **errp)
26{
27 VirtIOGPUPCI *vgpu = VIRTIO_GPU_PCI(vpci_dev);
28 VirtIOGPU *g = &vgpu->vdev;
29 DeviceState *vdev = DEVICE(&vgpu->vdev);
30 int i;
31
32 qdev_set_parent_bus(vdev, BUS(&vpci_dev->bus));
33 virtio_pci_force_virtio_1(vpci_dev);
34 object_property_set_bool(OBJECT(vdev), true, "realized", errp);
35
36 for (i = 0; i < g->conf.max_outputs; i++) {
37 object_property_set_link(OBJECT(g->scanout[i].con),
38 OBJECT(vpci_dev),
39 "device", errp);
40 }
41}
42
43static void virtio_gpu_pci_class_init(ObjectClass *klass, void *data)
44{
45 DeviceClass *dc = DEVICE_CLASS(klass);
46 VirtioPCIClass *k = VIRTIO_PCI_CLASS(klass);
47 PCIDeviceClass *pcidev_k = PCI_DEVICE_CLASS(klass);
48
49 set_bit(DEVICE_CATEGORY_DISPLAY, dc->categories);
50 dc->props = virtio_gpu_pci_properties;
51 dc->hotpluggable = false;
52 k->realize = virtio_gpu_pci_realize;
53 pcidev_k->class_id = PCI_CLASS_DISPLAY_OTHER;
54}
55
56static void virtio_gpu_initfn(Object *obj)
57{
58 VirtIOGPUPCI *dev = VIRTIO_GPU_PCI(obj);
59
60 virtio_instance_init_common(obj, &dev->vdev, sizeof(dev->vdev),
61 TYPE_VIRTIO_GPU);
62}
63
64static const TypeInfo virtio_gpu_pci_info = {
65 .name = TYPE_VIRTIO_GPU_PCI,
66 .parent = TYPE_VIRTIO_PCI,
67 .instance_size = sizeof(VirtIOGPUPCI),
68 .instance_init = virtio_gpu_initfn,
69 .class_init = virtio_gpu_pci_class_init,
70};
71
72static void virtio_gpu_pci_register_types(void)
73{
74 type_register_static(&virtio_gpu_pci_info);
75}
