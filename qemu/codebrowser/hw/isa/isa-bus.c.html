54ISABus *isa_bus_new(DeviceState *dev, MemoryRegion* address_space,
55 MemoryRegion *address_space_io, Error **errp)
56{
57 if (isabus) {
58 error_setg(errp, "Can't create a second ISA bus");
59 return NULL;
60 }
61 if (!dev) {
62 dev = qdev_create(NULL, "isabus-bridge");
63 qdev_init_nofail(dev);
64 }
65
66 isabus = ISA_BUS(qbus_create(TYPE_ISA_BUS, dev, NULL));
67 isabus->address_space = address_space;
68 isabus->address_space_io = address_space_io;
69 return isabus;
70}

77/*
78 * isa_get_irq() returns the corresponding qemu_irq entry for the i8259.
79 *
80 * This function is only for special cases such as the 'ferr', and
81 * temporary use for normal devices until they are converted to qdev.
82 */
83qemu_irq isa_get_irq(ISADevice *dev, int isairq)
84{
85 assert(!dev || ISA_BUS(qdev_get_parent_bus(DEVICE(dev))) == isabus);
86 if (isairq < 0 || isairq > 15) {
87 hw_error("isa irq %d invalid", isairq);
88 }
89 return isabus->irqs[isairq];
90}
91
92void isa_init_irq(ISADevice *dev, qemu_irq *p, int isairq)
93{
94 assert(dev->nirqs < ARRAY_SIZE(dev->isairq));
95 dev->isairq[dev->nirqs] = isairq;
96 *p = isa_get_irq(dev, isairq);
97 dev->nirqs++;
98}

100void isa_connect_gpio_out(ISADevice *isadev, int gpioirq, int isairq)
101{
102 qemu_irq irq;
103 isa_init_irq(isadev, &irq, isairq);
104 qdev_connect_gpio_out(DEVICE(isadev), gpioirq, irq);
105}

107void isa_bus_dma(ISABus *bus, IsaDma *dma8, IsaDma *dma16)
108{
109 assert(bus && dma8 && dma16);
110 assert(!bus->dma[0] && !bus->dma[1]);
111 bus->dma[0] = dma8;
112 bus->dma[1] = dma16;
113}
114
115IsaDma *isa_get_dma(ISABus *bus, int nchan)
116{
117 assert(bus);
118 return bus->dma[nchan > 3 ? 1 : 0];
119}

121static inline void isa_init_ioport(ISADevice *dev, uint16_t ioport)
122{
123 if (dev && (dev->ioport_id == 0 || ioport < dev->ioport_id)) {
124 dev->ioport_id = ioport;
125 }
126}

128void isa_register_ioport(ISADevice *dev, MemoryRegion *io, uint16_t start)
129{
130 memory_region_add_subregion(isabus->address_space_io, start, io);
131 isa_init_ioport(dev, start);
132}

134void isa_register_portio_list(ISADevice *dev,
135 PortioList *piolist, uint16_t start,
136 const MemoryRegionPortio *pio_start,
137 void *opaque, const char *name)
138{
139 assert(piolist && !piolist->owner);
140
141 /* START is how we should treat DEV, regardless of the actual
142 contents of the portio array. This is how the old code
143 actually handled e.g. the FDC device. */
144 isa_init_ioport(dev, start);
145
146 portio_list_init(piolist, OBJECT(dev), pio_start, opaque, name);
147 portio_list_add(piolist, isabus->address_space_io, start);
148}

183ISADevice *isa_vga_init(ISABus *bus)
184{
185 switch (vga_interface_type) {
186 case VGA_CIRRUS:
187 return isa_create_simple(bus, "isa-cirrus-vga");
188 case VGA_QXL:
189 fprintf(stderr, "%s: qxl: no PCI bus\n", __func__);
190 return NULL;
191 case VGA_STD:
192 return isa_create_simple(bus, "isa-vga");
193 case VGA_VMWARE:
194 fprintf(stderr, "%s: vmware_vga: no PCI bus\n", __func__);
195 return NULL;
196 case VGA_VIRTIO:
197 fprintf(stderr, "%s: virtio-vga: no PCI bus\n", __func__);
198 return NULL;
199 case VGA_NONE:
200 default:
201 return NULL;
202 }
203}

205static void isabus_dev_print(Monitor *mon, DeviceState *dev, int indent)
206{
207 ISADevice *d = ISA_DEVICE(dev);
208
209 if (d->isairq[1] != -1) {
210 monitor_printf(mon, "%*sisa irqs %d,%d\n", indent, "",
211 d->isairq[0], d->isairq[1]);
212 } else if (d->isairq[0] != -1) {
213 monitor_printf(mon, "%*sisa irq %d\n", indent, "",
214 d->isairq[0]);
215 }
216}

257static char *isabus_get_fw_dev_path(DeviceState *dev)
258{
259 ISADevice *d = ISA_DEVICE(dev);
260 char path[40];
261 int off;
262
263 off = snprintf(path, sizeof(path), "%s", qdev_fw_name(dev));
264 if (d->ioport_id) {
265 snprintf(path + off, sizeof(path) - off, "@%04x", d->ioport_id);
266 }
267
268 return g_strdup(path);
269}

291static void parallel_init(ISABus *bus, int index, Chardev *chr)
292{
293 DeviceState *dev;
294 ISADevice *isadev;
295
296 isadev = isa_create(bus, "isa-parallel");
297 dev = DEVICE(isadev);
298 qdev_prop_set_uint32(dev, "index", index);
299 qdev_prop_set_chr(dev, "chardev", chr);
300 qdev_init_nofail(dev);
301}

303void parallel_hds_isa_init(ISABus *bus, int n)
304{
305 int i;
306
307 assert(n <= MAX_PARALLEL_PORTS);
308
309 for (i = 0; i < n; i++) {
310 if (parallel_hds[i]) {
311 parallel_init(bus, i, parallel_hds[i]);
312 }
313 }
