27#include "qemu/osdep.h"
28#include "hw/pci/pci.h"
29#include "hw/i386/pc.h"
30#include "hw/loader.h"
31#include "sysemu/sysemu.h"
32
33#define SGABIOS_FILENAME "sgabios.bin"
34
35#define TYPE_SGA "sga"
36#define SGA(obj) OBJECT_CHECK(ISASGAState, (obj), TYPE_SGA)
37
38typedef struct ISASGAState {
39 ISADevice parent_obj;
40} ISASGAState;
41
42static void sga_realizefn(DeviceState *dev, Error **errp)
43{
44 rom_add_vga(SGABIOS_FILENAME);
45}
46
47static void sga_class_initfn(ObjectClass *klass, void *data)
48{
49 DeviceClass *dc = DEVICE_CLASS(klass);
50
51 set_bit(DEVICE_CATEGORY_DISPLAY, dc->categories);
52 dc->realize = sga_realizefn;
53 dc->desc = "Serial Graphics Adapter";
54}
55
56static const TypeInfo sga_info = {
57 .name = TYPE_SGA,
58 .parent = TYPE_ISA_DEVICE,
59 .instance_size = sizeof(ISASGAState),
60 .class_init = sga_class_initfn,
61};
62
63static void sga_register_types(void)
64{
65 type_register_static(&sga_info);
66}
67
68type_init(sga_register_types)
