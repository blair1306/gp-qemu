16void hotplug_handler_pre_plug(HotplugHandler *plug_handler,
17 DeviceState *plugged_dev,
18 Error **errp)
19{
20 HotplugHandlerClass *hdc = HOTPLUG_HANDLER_GET_CLASS(plug_handler);
21
22 if (hdc->pre_plug) {
23 hdc->pre_plug(plug_handler, plugged_dev, errp);
24 }
25}
