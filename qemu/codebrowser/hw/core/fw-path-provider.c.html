18#include "qemu/osdep.h"
19#include "hw/fw-path-provider.h"
20
21char *fw_path_provider_get_dev_path(FWPathProvider *p, BusState *bus,
22 DeviceState *dev)
23{
24 FWPathProviderClass *k = FW_PATH_PROVIDER_GET_CLASS(p);
25
26 return k->get_dev_path(p, bus, dev);
27}
28
29char *fw_path_provider_try_get_dev_path(Object *o, BusState *bus,
30 DeviceState *dev)
31{
32 FWPathProvider *p = (FWPathProvider *)
33 object_dynamic_cast(o, TYPE_FW_PATH_PROVIDER);
34
35 if (p) {
36 return fw_path_provider_get_dev_path(p, bus, dev);
37 }
38
39 return NULL;
40}
41
42static const TypeInfo fw_path_provider_info = {
43 .name = TYPE_FW_PATH_PROVIDER,
44 .parent = TYPE_INTERFACE,
45 .class_size = sizeof(FWPathProviderClass),
46};
47
48static void fw_path_provider_register_types(void)
49{
50 type_register_static(&fw_path_provider_info);
51}
52
53type_init(fw_path_provider_register_types)
