257/* DMA Remapping Fault Conditions */
258typedef enum VTDFaultReason {
259 VTD_FR_RESERVED = 0, /* Reserved for Advanced Fault logging */
260 VTD_FR_ROOT_ENTRY_P = 1, /* The Present(P) field of root-entry is 0 */
261 VTD_FR_CONTEXT_ENTRY_P, /* The Present(P) field of context-entry is 0 */
262 VTD_FR_CONTEXT_ENTRY_INV, /* Invalid programming of a context-entry */
263 VTD_FR_ADDR_BEYOND_MGAW, /* Input-address above (2^x-1) */
264 VTD_FR_WRITE, /* No write permission */
265 VTD_FR_READ, /* No read permission */
266 /* Fail to access a second-level paging entry (not SL_PML4E) */
267 VTD_FR_PAGING_ENTRY_INV,
268 VTD_FR_ROOT_TABLE_INV, /* Fail to access a root-entry */
269 VTD_FR_CONTEXT_TABLE_INV, /* Fail to access a context-entry */
270 /* Non-zero reserved field in a present root-entry */
271 VTD_FR_ROOT_ENTRY_RSVD,
272 /* Non-zero reserved field in a present context-entry */
273 VTD_FR_CONTEXT_ENTRY_RSVD,
274 /* Non-zero reserved field in a second-level paging entry with at lease one
275 * Read(R) and Write(W) or Execute(E) field is Set.
276 */
277 VTD_FR_PAGING_ENTRY_RSVD,
278 /* Translation request or translated request explicitly blocked dut to the
279 * programming of the Translation Type (T) field in the present
280 * context-entry.
281 */
282 VTD_FR_CONTEXT_ENTRY_TT,
283
284 /* Interrupt remapping transition faults */
285 VTD_FR_IR_REQ_RSVD = 0x20, /* One or more IR request reserved
286 * fields set */
287 VTD_FR_IR_INDEX_OVER = 0x21, /* Index value greater than max */
288 VTD_FR_IR_ENTRY_P = 0x22, /* Present (P) not set in IRTE */
289 VTD_FR_IR_ROOT_INVAL = 0x23, /* IR Root table invalid */
290 VTD_FR_IR_IRTE_RSVD = 0x24, /* IRTE Rsvd field non-zero with
291 * Present flag set */
292 VTD_FR_IR_REQ_COMPAT = 0x25, /* Encountered compatible IR
293 * request while disabled */
294 VTD_FR_IR_SID_ERR = 0x26, /* Invalid Source-ID */
295
296 /* This is not a normal fault reason. We use this to indicate some faults
297 * that are not referenced by the VT-d specification.
298 * Fault event with such reason should not be recorded.
299 */
300 VTD_FR_RESERVED_ERR,
301 VTD_FR_MAX, /* Guard */
302} VTDFaultReason;
