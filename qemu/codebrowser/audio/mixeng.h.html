25#ifndef QEMU_MIXENG_H
26#define QEMU_MIXENG_H
27
28#ifdef FLOAT_MIXENG
29typedef float mixeng_real;
30struct mixeng_volume { int mute; mixeng_real r; mixeng_real l; };
31struct st_sample { mixeng_real l; mixeng_real r; };
32#else
33struct mixeng_volume { int mute; int64_t r; int64_t l; };
34struct st_sample { int64_t l; int64_t r; };
35#endif
36
37typedef void (t_sample) (struct st_sample *dst, const void *src, int samples);
38typedef void (f_sample) (void *dst, const struct st_sample *src, int samples);
39
40extern t_sample *mixeng_conv[2][2][2][3];
41extern f_sample *mixeng_clip[2][2][2][3];
42
43void *st_rate_start (int inrate, int outrate);
44void st_rate_flow (void *opaque, struct st_sample *ibuf, struct st_sample *obuf,
45 int *isamp, int *osamp);
46void st_rate_flow_mix (void *opaque, struct st_sample *ibuf, struct st_sample *obuf,
47 int *isamp, int *osamp);
48void st_rate_stop (void *opaque);
49void mixeng_clear (struct st_sample *buf, int len);
50void mixeng_volume (struct st_sample *buf, int len, struct mixeng_volume *vol);
51
52#endif /* QEMU_MIXENG_H */
