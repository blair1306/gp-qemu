140static inline void *advance (void *p, int incr)
141{
142 uint8_t *d = p;
143 return (d + incr);
144}

146#ifdef __GNUC__
147#define audio_MIN(a, b) ( __extension__ ({ \
148 __typeof (a) ta = a; \
149 __typeof (b) tb = b; \
150 ((ta)>(tb)?(tb):(ta)); \
151}))
152
153#define audio_MAX(a, b) ( __extension__ ({ \
154 __typeof (a) ta = a; \
155 __typeof (b) tb = b; \
156 ((ta)<(tb)?(tb):(ta)); \
157}))
158#else
159#define audio_MIN(a, b) ((a)>(b)?(b):(a))
160#define audio_MAX(a, b) ((a)<(b)?(b):(a))
161#endif
