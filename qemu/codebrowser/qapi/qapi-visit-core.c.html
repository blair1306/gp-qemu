24void visit_complete(Visitor *v, void *opaque)
25{
26 assert(v->type != VISITOR_OUTPUT || v->complete);
27 trace_visit_complete(v, opaque);
28 if (v->complete) {
29 v->complete(v, opaque);
30 }
31}

41void visit_start_struct(Visitor *v, const char *name, void **obj,
42 size_t size, Error **errp)
43{
44 Error *err = NULL;
45
46 trace_visit_start_struct(v, name, obj, size);
47 if (obj) {
48 assert(size);
49 assert(!(v->type & VISITOR_OUTPUT) || *obj);
50 }
51 v->start_struct(v, name, obj, size, &err);
52 if (obj && (v->type & VISITOR_INPUT)) {
53 assert(!err != !*obj);
54 }
55 error_propagate(errp, err);
56}

72void visit_start_list(Visitor *v, const char *name, GenericList **list,
73 size_t size, Error **errp)
74{
75 Error *err = NULL;
76
77 assert(!list || size >= sizeof(GenericList));
78 trace_visit_start_list(v, name, list, size);
79 v->start_list(v, name, list, size, &err);
80 if (list && (v->type & VISITOR_INPUT)) {
81 assert(!(err && *list));
82 }
83 error_propagate(errp, err);
84}

107void visit_start_alternate(Visitor *v, const char *name,
108 GenericAlternate **obj, size_t size,
109 Error **errp)
110{
111 Error *err = NULL;
112
113 assert(obj && size >= sizeof(GenericAlternate));
114 assert(!(v->type & VISITOR_OUTPUT) || *obj);
115 trace_visit_start_alternate(v, name, obj, size);
116 if (v->start_alternate) {

154static void visit_type_uintN(Visitor *v, uint64_t *obj, const char *name,
155 uint64_t max, const char *type, Error **errp)
156{
157 Error *err = NULL;
158 uint64_t value = *obj;
159
160 v->type_uint64(v, name, &value, &err);
161 if (err) {
162 error_propagate(errp, err);
163 } else if (value > max) {
164 error_setg(errp, QERR_INVALID_PARAMETER_VALUE,
165 name ? name : "null", type);
166 } else {
167 *obj = value;
168 }
169}

335static void output_type_enum(Visitor *v, const char *name, int *obj,
336 const char *const strings[], Error **errp)
337{
338 int i = 0;
339 int value = *obj;
340 char *enum_str;
341
342 while (strings[i++] != NULL);
343 if (value < 0 || value >= i - 1) {
344 error_setg(errp, QERR_INVALID_PARAMETER, name ? name : "null");
345 return;
346 }
347
348 enum_str = (char *)strings[value];
349 visit_type_str(v, name, &enum_str, errp);
350}
351
352static void input_type_enum(Visitor *v, const char *name, int *obj,
353 const char *const strings[], Error **errp)
354{
355 Error *local_err = NULL;
356 int64_t value = 0;
357 char *enum_str;
358
359 visit_type_str(v, name, &enum_str, &local_err);
360 if (local_err) {
361 error_propagate(errp, local_err);
362 return;
363 }
364
365 while (strings[value] != NULL) {
366 if (strcmp(strings[value], enum_str) == 0) {
367 break;
368 }
369 value++;
370 }
371
372 if (strings[value] == NULL) {
373 error_setg(errp, QERR_INVALID_PARAMETER, enum_str);
374 g_free(enum_str);
375 return;
376 }
377
378 g_free(enum_str);
379 *obj = value;
380}

382void visit_type_enum(Visitor *v, const char *name, int *obj,
383 const char *const strings[], Error **errp)
384{
385 assert(obj && strings);
386 trace_visit_type_enum(v, name, obj);
387 switch (v->type) {
388 case VISITOR_INPUT:
389 input_type_enum(v, name, obj, strings, errp);
390 break;
391 case VISITOR_OUTPUT:
392 output_type_enum(v, name, obj, strings, errp);
393 break;
394 case VISITOR_CLONE:
395 /* nothing further to do, scalar value was already copied by
396 * g_memdup() during visit_start_*() */
397 break;
398 case VISITOR_DEALLOC:
399 /* nothing to deallocate for a scalar */
400 break;
401 }
