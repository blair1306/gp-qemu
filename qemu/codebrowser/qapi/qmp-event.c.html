33static void timestamp_put(QDict *qdict)
34{
35 int err;
36 QObject *obj;
37 qemu_timeval tv;
38
39 err = qemu_gettimeofday(&tv);
40 /* Put -1 to indicate failure of getting host time */
41 obj = qobject_from_jsonf("{ 'seconds': %lld, 'microseconds': %lld }",
42 err < 0 ? -1LL : (long long)tv.tv_sec,
43 err < 0 ? -1LL : (long long)tv.tv_usec);
44 qdict_put_obj(qdict, "timestamp", obj);
45}
46
47/*
48 * Build a QDict, then fill event name and time stamp, caller should free the
49 * QDict after usage.
50 */
51QDict *qmp_event_build_dict(const char *event_name)
52{
53 QDict *dict = qdict_new();
54 qdict_put_str(dict, "event", event_name);
55 timestamp_put(dict);
56 return dict;
57}
