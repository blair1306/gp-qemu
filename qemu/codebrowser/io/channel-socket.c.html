48QIOChannelSocket *
49qio_channel_socket_new(void)
50{
51 QIOChannelSocket *sioc;
52 QIOChannel *ioc;
53
54 sioc = QIO_CHANNEL_SOCKET(object_new(TYPE_QIO_CHANNEL_SOCKET));
55 sioc->fd = -1;
56
57 ioc = QIO_CHANNEL(sioc);
58 qio_channel_set_feature(ioc, QIO_CHANNEL_FEATURE_SHUTDOWN);
59
60#ifdef WIN32
61 ioc->event = CreateEvent(NULL, FALSE, FALSE, NULL);
62#endif
63
64 trace_qio_channel_socket_new(sioc);
65
66 return sioc;
67}

70static int
71qio_channel_socket_set_fd(QIOChannelSocket *sioc,
72 int fd,
73 Error **errp)
74{
75 if (sioc->fd != -1) {
76 error_setg(errp, "Socket is already open");
77 return -1;
78 }
79
80 sioc->fd = fd;
81 sioc->remoteAddrLen = sizeof(sioc->remoteAddr);
82 sioc->localAddrLen = sizeof(sioc->localAddr);
83
84
85 if (getpeername(fd, (struct sockaddr *)&sioc->remoteAddr,
86 &sioc->remoteAddrLen) < 0) {
87 if (errno == ENOTCONN) {
88 memset(&sioc->remoteAddr, 0, sizeof(sioc->remoteAddr));
89 sioc->remoteAddrLen = sizeof(sioc->remoteAddr);
90 } else {
91 error_setg_errno(errp, errno,
92 "Unable to query remote socket address");
93 goto error;
94 }
95 }
96
97 if (getsockname(fd, (struct sockaddr *)&sioc->localAddr,
98 &sioc->localAddrLen) < 0) {
99 error_setg_errno(errp, errno,
100 "Unable to query local socket address");
101 goto error;
102 }
103
104#ifndef WIN32
105 if (sioc->localAddr.ss_family == AF_UNIX) {
106 QIOChannel *ioc = QIO_CHANNEL(sioc);
107 qio_channel_set_feature(ioc, QIO_CHANNEL_FEATURE_FD_PASS);
108 }
109#endif /* WIN32 */
110
111 return 0;
112
113 error:
114 sioc->fd = -1; /* Let the caller close FD on failure */
115 return -1;
116}

136int qio_channel_socket_connect_sync(QIOChannelSocket *ioc,
137 SocketAddress *addr,
138 Error **errp)
139{
140 int fd;
141
142 trace_qio_channel_socket_connect_sync(ioc, addr);
143 fd = socket_connect(addr, NULL, NULL, errp);
144 if (fd < 0) {
145 trace_qio_channel_socket_connect_fail(ioc);
146 return -1;
147 }
148
149 trace_qio_channel_socket_connect_complete(ioc, fd);
150 if (qio_channel_socket_set_fd(ioc, fd, errp) < 0) {
151 close(fd);
152 return -1;
153 }
154
155 return 0;
156}

172void qio_channel_socket_connect_async(QIOChannelSocket *ioc,
173 SocketAddress *addr,
174 QIOTaskFunc callback,
175 gpointer opaque,
176 GDestroyNotify destroy)
177{
178 QIOTask *task = qio_task_new(
179 OBJECT(ioc), callback, opaque, destroy);
180 SocketAddress *addrCopy;
181
182 addrCopy = QAPI_CLONE(SocketAddress, addr);
183
184 /* socket_connect() does a non-blocking connect(), but it
185 * still blocks in DNS lookups, so we must use a thread */
186 trace_qio_channel_socket_connect_async(ioc, addr);
187 qio_task_run_in_thread(task,
188 qio_channel_socket_connect_worker,
189 addrCopy,
190 (GDestroyNotify)qapi_free_SocketAddress);
191}

328QIOChannelSocket *
329qio_channel_socket_accept(QIOChannelSocket *ioc,
330 Error **errp)
331{
332 QIOChannelSocket *cioc;
333
334 cioc = qio_channel_socket_new();
335 cioc->remoteAddrLen = sizeof(ioc->remoteAddr);
336 cioc->localAddrLen = sizeof(ioc->localAddr);
337
338 retry:
339 trace_qio_channel_socket_accept(ioc);
340 cioc->fd = qemu_accept(ioc->fd, (struct sockaddr *)&cioc->remoteAddr,
341 &cioc->remoteAddrLen);
342 if (cioc->fd < 0) {
343 trace_qio_channel_socket_accept_fail(ioc);
344 if (errno == EINTR) {
345 goto retry;
346 }
347 goto error;
348 }
349
350 if (getsockname(cioc->fd, (struct sockaddr *)&cioc->localAddr,
351 &cioc->localAddrLen) < 0) {
352 error_setg_errno(errp, errno,
353 "Unable to query local socket address");
354 goto error;
355 }
356
357#ifndef WIN32
358 if (cioc->localAddr.ss_family == AF_UNIX) {
359 QIOChannel *ioc_local = QIO_CHANNEL(cioc);
360 qio_channel_set_feature(ioc_local, QIO_CHANNEL_FEATURE_FD_PASS);
361 }
362#endif /* WIN32 */
363
364 trace_qio_channel_socket_accept_complete(ioc, cioc, cioc->fd);
365 return cioc;
366
367 error:
368 object_unref(OBJECT(cioc));
369 return NULL;
370}

378static void qio_channel_socket_finalize(Object *obj)
379{
380 QIOChannelSocket *ioc = QIO_CHANNEL_SOCKET(obj);
381
382 if (ioc->fd != -1) {
383 QIOChannel *ioc_local = QIO_CHANNEL(ioc);
384 if (qio_channel_has_feature(ioc_local, QIO_CHANNEL_FEATURE_LISTEN)) {
385 Error *err = NULL;
386
387 socket_listen_cleanup(ioc->fd, &err);
388 if (err) {
389 error_report_err(err);
390 err = NULL;
391 }
392 }
393#ifdef WIN32
394 WSAEventSelect(ioc->fd, NULL, 0);
395#endif
396 closesocket(ioc->fd);
397 ioc->fd = -1;
398 }
399}

402#ifndef WIN32
403static void qio_channel_socket_copy_fds(struct msghdr *msg,
404 int **fds, size_t *nfds)
405{
406 struct cmsghdr *cmsg;
407
408 *nfds = 0;
409 *fds = NULL;
410
411 for (cmsg = CMSG_FIRSTHDR(msg); cmsg; cmsg = CMSG_NXTHDR(msg, cmsg)) {
412 int fd_size, i;
413 int gotfds;
414
415 if (cmsg->cmsg_len < CMSG_LEN(sizeof(int)) ||
416 cmsg->cmsg_level != SOL_SOCKET ||
417 cmsg->cmsg_type != SCM_RIGHTS) {
418 continue;
419 }
420
421 fd_size = cmsg->cmsg_len - CMSG_LEN(0);
422
423 if (!fd_size) {
424 continue;
425 }
426
427 gotfds = fd_size / sizeof(int);
428 *fds = g_renew(int, *fds, *nfds + gotfds);
429 memcpy(*fds + *nfds, CMSG_DATA(cmsg), fd_size);
430
431 for (i = 0; i < gotfds; i++) {
432 int fd = (*fds)[*nfds + i];
433 if (fd < 0) {
434 continue;
435 }
436
437 /* O_NONBLOCK is preserved across SCM_RIGHTS so reset it */
438 qemu_set_block(fd);
439
440#ifndef MSG_CMSG_CLOEXEC
441 qemu_set_cloexec(fd);
442#endif
443 }
444 *nfds += gotfds;
445 }
446}

449static ssize_t qio_channel_socket_readv(QIOChannel *ioc,
450 const struct iovec *iov,
451 size_t niov,
452 int **fds,
453 size_t *nfds,
454 Error **errp)
455{
456 QIOChannelSocket *sioc = QIO_CHANNEL_SOCKET(ioc);
457 ssize_t ret;
458 struct msghdr msg = { NULL, };
459 char control[CMSG_SPACE(sizeof(int) * SOCKET_MAX_FDS)];
460 int sflags = 0;
461
462 memset(control, 0, CMSG_SPACE(sizeof(int) * SOCKET_MAX_FDS));
463
464#ifdef MSG_CMSG_CLOEXEC
465 sflags |= MSG_CMSG_CLOEXEC;
466#endif
467
468 msg.msg_iov = (struct iovec *)iov;
469 msg.msg_iovlen = niov;
470 if (fds && nfds) {
471 msg.msg_control = control;
472 msg.msg_controllen = sizeof(control);
473 }
474
475 retry:
476 ret = recvmsg(sioc->fd, &msg, sflags);
477 if (ret < 0) {
478 if (errno == EAGAIN) {
479 return QIO_CHANNEL_ERR_BLOCK;
480 }
481 if (errno == EINTR) {
482 goto retry;
483 }
484
485 error_setg_errno(errp, errno,
486 "Unable to read from socket");
487 return -1;
488 }
489
490 if (fds && nfds) {
491 qio_channel_socket_copy_fds(&msg, fds, nfds);
492 }
493
494 return ret;
495}

497static ssize_t qio_channel_socket_writev(QIOChannel *ioc,
498 const struct iovec *iov,
499 size_t niov,
500 int *fds,
501 size_t nfds,
502 Error **errp)
503{
504 QIOChannelSocket *sioc = QIO_CHANNEL_SOCKET(ioc);
505 ssize_t ret;
506 struct msghdr msg = { NULL, };
507 char control[CMSG_SPACE(sizeof(int) * SOCKET_MAX_FDS)];
508 size_t fdsize = sizeof(int) * nfds;
509 struct cmsghdr *cmsg;
510
511 memset(control, 0, CMSG_SPACE(sizeof(int) * SOCKET_MAX_FDS));
512
513 msg.msg_iov = (struct iovec *)iov;
514 msg.msg_iovlen = niov;
515
516 if (nfds) {
517 if (nfds > SOCKET_MAX_FDS) {
518 error_setg_errno(errp, EINVAL,
519 "Only %d FDs can be sent, got %zu",
520 SOCKET_MAX_FDS, nfds);
521 return -1;
522 }
523
524 msg.msg_control = control;
525 msg.msg_controllen = CMSG_SPACE(sizeof(int) * nfds);
526
527 cmsg = CMSG_FIRSTHDR(&msg);
528 cmsg->cmsg_len = CMSG_LEN(fdsize);
529 cmsg->cmsg_level = SOL_SOCKET;
530 cmsg->cmsg_type = SCM_RIGHTS;
531 memcpy(CMSG_DATA(cmsg), fds, fdsize);
532 }
533
534 retry:
535 ret = sendmsg(sioc->fd, &msg, 0);
536 if (ret <= 0) {
537 if (errno == EAGAIN) {
538 return QIO_CHANNEL_ERR_BLOCK;
539 }
540 if (errno == EINTR) {
541 goto retry;
542 }
543 error_setg_errno(errp, errno,
544 "Unable to write to socket");
545 return -1;
546 }
547 return ret;
548}
549#else /* WIN32 */
550static ssize_t qio_channel_socket_readv(QIOChannel *ioc,
551 const struct iovec *iov,
552 size_t niov,
553 int **fds,
554 size_t *nfds,
555 Error **errp)
556{
557 QIOChannelSocket *sioc = QIO_CHANNEL_SOCKET(ioc);
558 ssize_t done = 0;
559 ssize_t i;
560
561 for (i = 0; i < niov; i++) {
562 ssize_t ret;
563 retry:
564 ret = recv(sioc->fd,
565 iov[i].iov_base,
566 iov[i].iov_len,
567 0);
568 if (ret < 0) {
569 if (errno == EAGAIN) {
570 if (done) {
571 return done;
572 } else {
573 return QIO_CHANNEL_ERR_BLOCK;
574 }
575 } else if (errno == EINTR) {
576 goto retry;
577 } else {
578 error_setg_errno(errp, errno,
579 "Unable to read from socket");
580 return -1;
581 }
582 }
583 done += ret;
584 if (ret < iov[i].iov_len) {
585 return done;
586 }
587 }
588
589 return done;
590}

592static ssize_t qio_channel_socket_writev(QIOChannel *ioc,
593 const struct iovec *iov,
594 size_t niov,
595 int *fds,
596 size_t nfds,

611 if (errno == EAGAIN) {
612 if (done) {
613 return done;
614 } else {
615 return QIO_CHANNEL_ERR_BLOCK;
616 }
617 } else if (errno == EINTR) {
618 goto retry;
619 } else {
620 error_setg_errno(errp, errno,
621 "Unable to write to socket");
622 return -1;
623 }
624 }
625 done += ret;
626 if (ret < iov[i].iov_len) {
627 return done;
628 }
629 }
630
631 return done;

696static int
697qio_channel_socket_shutdown(QIOChannel *ioc,
698 QIOChannelShutdown how,
699 Error **errp)
700{
701 QIOChannelSocket *sioc = QIO_CHANNEL_SOCKET(ioc);
702 int sockhow;
703
704 switch (how) {
705 case QIO_CHANNEL_SHUTDOWN_READ:
706 sockhow = SHUT_RD;
707 break;
708 case QIO_CHANNEL_SHUTDOWN_WRITE:
709 sockhow = SHUT_WR;
710 break;
711 case QIO_CHANNEL_SHUTDOWN_BOTH:
712 default:
713 sockhow = SHUT_RDWR;
714 break;
715 }
716
717 if (shutdown(sioc->fd, sockhow) < 0) {
718 error_setg_errno(errp, errno,
719 "Unable to shutdown socket");
720 return -1;
721 }
722 return 0;
723}
