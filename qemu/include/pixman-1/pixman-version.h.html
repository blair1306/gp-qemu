40#define PIXMAN_VERSION_ENCODE(major, minor, micro) ( \
41 ((major) * 10000) \
42 + ((minor) * 100) \
43 + ((micro) * 1))
44
45#define PIXMAN_VERSION PIXMAN_VERSION_ENCODE( \
