126/*
127 * This function is used to parse an iSCSI URL into a iscsi_url structure.
128 * iSCSI URL format :
129 * iscsi://[<username>[%<password>]@]<host>[:<port>]/<target-iqn>/<lun>
130 *
131 * Target names are url encoded with '%' as a special character.
132 * Example:
133 * "iqn.ronnie.test%3A1234" will be translated to "iqn.ronnie.test:1234"
134 *
135 * Function will return a pointer to an iscsi url structure if successful,
136 * or it will return NULL and set iscsi_get_error() accrodingly if there was a problem
137 * with the URL.
138 *
139 * CHAP username/password can also be provided via environment variables
140 * LIBISCSI_CHAP_USERNAME=ronnie
141 * LIBISCSI_CHAP_PASSWORD=password
142 *
143 * The returned structure is freed by calling iscsi_destroy_url()
144 */
145EXTERN struct iscsi_url *iscsi_parse_full_url(struct iscsi_context *iscsi, const char *url);
146EXTERN void iscsi_destroy_url(struct iscsi_url *iscsi_url);
147
148/*
149 * This function is used to parse an iSCSI Portal URL into a iscsi_url structure.
150 * iSCSI Portal URL format :
151 * iscsi://[<username>[%<password>]@]<host>[:<port>]
152 *
153 * iSCSI Portal URL is used when describing a discovery session, where the target-iqn and the
154 * lun is not yet known.
155 *
156 * Function will return a pointer to an iscsi url structure if successful,
157 * or it will return NULL and set iscsi_get_error() accrodingly if there was a problem
158 * with the URL.
159 *
160 * CHAP username/password can also be provided via environment variables
161 * LIBISCSI_CHAP_USERNAME=ronnie
162 * LIBISCSI_CHAP_PASSWORD=password
163 *
164 * The returned structure is freed by calling iscsi_destroy_url()
165 */
166EXTERN struct iscsi_url *
167iscsi_parse_portal_url(struct iscsi_context *iscsi, const char *url);

303/*
304 * Asynchronous call to connect a TCP connection to the target-host/port
305 *
306 * Returns:
307 * 0 if the call was initiated and a connection will be attempted. Result of
308 * the connection will be reported through the callback function.
309 * <0 if there was an error. The callback function will not be invoked.
310 *
311 * This command is unique in that the callback can be invoked twice.
312 *
313 * Callback parameters :
314 * status can be either of :
315 * ISCSI_STATUS_GOOD : Connection was successful. Command_data is NULL.
316 * In this case the callback will be invoked a
317 * second time once the connection is torn down.
318 *
319 * ISCSI_STATUS_ERROR : Either failed to establish the connection, or
320 * an already established connection has failed
321 * with an error.
322 *
323 * The callback will NOT be invoked if the session is explicitely torn down
324 * through a call to iscsi_disconnect() or iscsi_destroy_context().
325 */
326EXTERN int iscsi_connect_async(struct iscsi_context *iscsi, const char *portal,
327 iscsi_command_cb cb, void *private_data);
328
329/*
330 * Synchronous call to connect a TCP connection to the target-host/port

332 * Returns:
333 * 0 if connected successfully.
334 * <0 if there was an error.

336 */

528 * Asynchronous call for task management
529 *
530 * Returns:
531 * 0 if the call was initiated and the task mgmt function will be invoked.
532 * the connection will be reported through the callback function.
533 * <0 if there was an error. The callback function will not be invoked.
534 *
535 * Callback parameters :
536 * status can be either of :
537 * ISCSI_STATUS_GOOD : Connection was successful. Command_data is a pointer to uint32_t

542 * The callback will NOT be invoked if the session is explicitely torn down

545 * abort_task will also cancel the scsi task. The callback for the scsi task will be invoked with
546 * SCSI_STATUS_CANCELLED

548 * all tasks will be invoked with SCSI_STATUS_CANCELLED
549 */

551iscsi_task_mgmt_async(struct iscsi_context *iscsi,
552 int lun, enum iscsi_task_mgmt_funcs function,

554 iscsi_command_cb cb, void *private_data);

607/* These are the possible status values for the callbacks for scsi commands.
608 * The content of command_data depends on the status type.
609 *
610 * status :
611 * ISCSI_STATUS_GOOD the scsi command completed successfullt on the target.
612 * If this scsi command returns DATA-IN, that data is stored in an scsi_task
613 * structure returned in the command_data parameter. This buffer will be
614 * automatically freed once the callback returns.
615 *
616 * ISCSI_STATUS_CHECK_CONDITION the scsi command failed with a scsi sense.
617 * Command_data contains a struct scsi_task. When the callback returns,
618 * this buffer will automatically become freed.
619 *
620 * ISCSI_STATUS_CANCELLED the scsi command was aborted. Command_data is
621 * NULL.
622 *
623 * ISCSI_STATUS_ERROR the command failed. Command_data is NULL.
624 */

627 size_t size;
628 unsigned char *data;
629};

1094/*
1095 * These functions are used when the application wants to specify its own buffers to read the data
1096 * from the DATA-IN PDUs into, or write the data to DATA-OUT PDUs from.
1097 * The main use is for SCSI READ10/12/16 WRITE10/12/16 operations to have them read/write directly from
1098 * the applications buffer, avoiding coying the data.
1099 *
1100 * This also supports reading into a vector of buffers by calling this function multiple times.
1101 * The individual buffers will be filled in the same order as they were created.
1102 *
1103 * Example READ10:
1104 * task = iscsi_read10_task( ( 2 512byte blocks into two buffers)
1105 * scsi_task_add_data_in_buffer(task, first_buffer, 512
1106 * scsi_task_add_data_in_buffer(task, second_buffer, 512
1107 *
1108 *
1109 * If you use this function you can not use task->datain in the READ callback.
1110 * task->datain.size will be 0 and
1111 * task->datain.data will be NULL
1112 *
1113 * Example WRITE10: (write 2 blocks)
1114 * static struct scsi_iovec iov[2];
1115 *
1116 * task = iscsi_write10_task(iscsi, lun, 0, NULL, 512, 512, 0, 0, 0, 0, 0, callback, private_data);
1117 * iov[0].iov_base = first_buffer;
1118 * iov[0].iov_len = 512;
1119 * iov[1].iov_base = second_buffer;
1120 * iov[1].iov_len = 512;
1121 * scsi_task_set_iov_out(task, &iov[0], 2);
1122 */
1123EXTERN int scsi_task_add_data_in_buffer(struct scsi_task *task, int len, unsigned char *buf);
1124EXTERN int scsi_task_add_data_out_buffer(struct scsi_task *task, int len, unsigned char *buf);
