129/*
130 * URL parsing functions.
131 * These functions all parse a URL of the form
132 * nfs://server/path/file?argv=val[&arg=val]*
133 * and returns a nfs_url.
134 *
135 * Apart from parsing the URL the functions will also update
136 * the nfs context to reflect settings controlled via url arguments.
137 *
138 * Current URL arguments are :
139 * tcp-syncnt=<int> : Number of SYNs to send during the seccion establish
140 * before failing settin up the tcp connection to the
141 * server.
142 * uid=<int> : UID value to use when talking to the server.
143 * default it 65534 on Windows and getuid() on unixen.
144 * gid=<int> : GID value to use when talking to the server.
145 * default it 65534 on Windows and getgid() on unixen.
146 * readahead=<int> : Enable readahead for files and set the maximum amount
147 * of readahead to <int>.
148 */
149/*
150 * Parse a complete NFS URL including, server, path and
151 * filename. Fail if any component is missing.
152 */
153EXTERN struct nfs_url *nfs_parse_url_full(struct nfs_context *nfs, const char *url);
154
155/*
156 * Parse an NFS URL, but do not split path and file. File
157 * in the resulting struct remains NULL.
158 */
159EXTERN struct nfs_url *nfs_parse_url_dir(struct nfs_context *nfs, const char *url);
160
161/*
162 * Parse an NFS URL, but do not fail if file, path or even server is missing.
163 * Check elements of the resulting struct for NULL.
164 */
165EXTERN struct nfs_url *nfs_parse_url_incomplete(struct nfs_context *nfs, const char *url);
166
167
168/*
169 * Free the URL struct returned by the nfs_parse_url_* functions.
170 */
171EXTERN void nfs_destroy_url(struct nfs_url *url);

400/*
401 * OPEN()
402 */
403/*
404 * Async open(<filename>)
405 *
406 * mode is a combination of the flags :
407 * O_RDOLNY, O_WRONLY, O_RDWR , O_SYNC, O_APPEND, O_TRUNC
408 *
409 * Function returns
410 * 0 : The operation was initiated. Once the operation finishes, the callback will be invoked.
411 * <0 : An error occured when trying to set up the operation. The callback will not be invoked.
412 *
413 * Supported flags are
414 * O_APPEND
415 * O_RDONLY
416 * O_WRONLY
417 * O_RDWR
418 * O_SYNC
419 * O_TRUNC (Only valid with O_RDWR or O_WRONLY. Ignored otherwise.)
420 *
421 * When the callback is invoked, status indicates the result:
422 * 0 : Success.
423 * data is a struct *nfsfh;
424 * The nfsfh is close using nfs_close().
425 * -errno : An error occured.
426 * data is the error string.
427 */
428EXTERN int nfs_open_async(struct nfs_context *nfs, const char *path, int flags, nfs_cb cb, void *private_data);
429/*
430 * Sync open(<filename>)
431 * Function returns
432 * 0 : The operation was successfull. *nfsfh is filled in.
433 * -errno : The command failed.
434 */
435EXTERN int nfs_open(struct nfs_context *nfs, const char *path, int flags, struct nfsfh **nfsfh);
